93// Main entry point for any SMACC state machine
94// It instantiates and starts the specified state machine type
95// it uses two threads: a new thread and the current one.
96// The created thread is for the state machine process
97// it locks the current thread to handle events of the state machine
98template <typename StateMachineType>
99void run()
100{
101 ::signal(SIGQUIT, onSigQuit);
102
103 // create the asynchronous state machine scheduler
104 SmaccFifoScheduler scheduler1(true);
105
106 // create the signalDetector component
107 SignalDetector signalDetector(&scheduler1);
108
109 // create the asynchronous state machine processor
110 SmaccFifoScheduler::processor_handle sm =
111 scheduler1.create_processor<StateMachineType>(&signalDetector);
112
113 // initialize the asynchronous state machine processor
114 signalDetector.setProcessorHandle(sm);
115
116 scheduler1.initiate_processor(sm);
117
118 //create a thread for the asynchronous state machine processor execution
119 boost::thread schedulerThread(boost::bind(&sc::fifo_scheduler<>::operator(), &scheduler1, 0));
120
121 // use the main thread for the signal detector component (waiting actionclient requests)
122 signalDetector.pollingLoop();
123}

134template <typename StateMachineType>
135SmExecution * run_async()
136{
137 ::signal(SIGQUIT, onSigQuit);
138
139 SmExecution * ret = new SmExecution();
140
141 // create the asynchronous state machine scheduler
142 ret->scheduler1 = new SmaccFifoScheduler(true);
143
144 // create the signalDetector component
145 ret->signalDetector = new SignalDetector(ret->scheduler1);
146
147 // create the asynchronous state machine processor
148 ret->sm = ret->scheduler1->create_processor<StateMachineType>(ret->signalDetector);
149
150 // initialize the asynchronous state machine processor
151 ret->signalDetector->setProcessorHandle(ret->sm);
152
153 ret->scheduler1->initiate_processor(ret->sm);
154
155 //create a thread for the asynchronous state machine processor execution
156 ret->schedulerThread =
157 new boost::thread(boost::bind(&sc::fifo_scheduler<>::operator(), ret->scheduler1, NULL));
158 ret->signalDetectorLoop =
159 new boost::thread(boost::bind(&SignalDetector::pollingLoop, ret->signalDetector));
160
161 return ret;
162}
