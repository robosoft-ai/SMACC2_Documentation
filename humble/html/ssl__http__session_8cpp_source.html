34void ssl_http_session::run(
35 const std::string & host, const std::string & target, const boost::beast::http::verb http_method,
36 const int & version)
37{
38 // Set up an HTTP request
39 req_.version(version);
40 req_.method(http_method);
41 req_.target(target);
42 req_.set(boost::beast::http::field::host, host);
43 req_.set(boost::beast::http::field::user_agent, BOOST_BEAST_VERSION_STRING);
44
45 // Look up the domain name
46 resolver_.async_resolve(
47 host.c_str(), kPort.c_str(),
48 boost::beast::bind_front_handler(&ssl_http_session::on_resolve, shared_from_this()));
49}

51void ssl_http_session::on_resolve(
52 boost::beast::error_code ec, boost::asio::ip::tcp::resolver::results_type results)
53{
54 if (ec) return fail(ec, "resolve");
55
56 // Set a timeout on the operation
57 boost::beast::get_lowest_layer(stream_).expires_after(std::chrono::seconds(1));
58
59 // Make the connection on the IP address we get from a lookup
60 boost::beast::get_lowest_layer(stream_).async_connect(
61 results, boost::beast::bind_front_handler(&ssl_http_session::on_connect, shared_from_this()));
62}

64void ssl_http_session::fail(boost::beast::error_code ec, char const * what)
65{
66 std::cout << "Failure!..." << std::endl;
67 std::cerr << what << ": " << ec.message() << "\n";
68 res_.result(boost::beast::http::status::bad_request);
69 res_.reason() = ec.message();
70 onResponse(res_);
71}

73void ssl_http_session::on_connect(
74 boost::beast::error_code ec, boost::asio::ip::tcp::resolver::results_type::endpoint_type)
75{
76 if (ec) return fail(ec, "connect");
77
78 // Set a timeout on the operation
79 boost::beast::get_lowest_layer(stream_).expires_after(std::chrono::seconds(1));
80
81 // Send the HTTP request to the remote host
82 stream_.async_handshake(
83 boost::asio::ssl::stream_base::client,
84 boost::beast::bind_front_handler(&ssl_http_session::on_handshake, shared_from_this()));
85}

87void ssl_http_session::on_handshake(boost::beast::error_code ec)
88{
89 if (ec) return fail(ec, "handshake");
90
91 // Set a timeout on the operation
92 boost::beast::get_lowest_layer(stream_).expires_after(std::chrono::seconds(1));
93
94 // Send the HTTP request to the remote host
95 boost::beast::http::async_write(
96 stream_, req_,
97 boost::beast::bind_front_handler(&ssl_http_session::on_write, shared_from_this()));
98}

100void ssl_http_session::on_write(boost::beast::error_code ec, std::size_t bytes_transferred)
101{
102 boost::ignore_unused(bytes_transferred);
103
104 if (ec) return fail(ec, "write");
105
106 // Receive the HTTP response
107 boost::beast::http::async_read(

109 boost::beast::bind_front_handler(&ssl_http_session::on_read, shared_from_this()));

112void ssl_http_session::on_read(boost::beast::error_code ec, std::size_t bytes_transferred)
113{
114 boost::ignore_unused(bytes_transferred);
115
116 if (ec) return fail(ec, "read");
117
118 boost::beast::get_lowest_layer(stream_).expires_after(std::chrono::seconds(1));
119
120 // Gracefully close the socket
121 // stream_.socket().shutdown(boost::asio::ip::tcp::socket::shutdown_both, ec);
122 stream_.async_shutdown(
123 boost::beast::bind_front_handler(&ssl_http_session::on_shutdown, shared_from_this()));
124}

126void ssl_http_session::on_shutdown(boost::beast::error_code ec)
127{
128 // not_connected happens sometimes so don't bother reporting it.
129 if (ec == boost::asio::error::eof)
130 {
131 // Rationale:
132 // http://stackoverflow.com/questions/25587403/boost-asio-ssl-async-shutdown-always-finishes-with-an-error
133 ec = {};
134 }
135 // if (ec) return fail(ec, "shutdown");
136
137 // If we get here then the connection is closed gracefully
138 onResponse(res_);
139}
