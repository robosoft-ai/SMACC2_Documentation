33void http_session::run(
34 const std::string & host, const std::string & target, const boost::beast::http::verb http_method,
35 const int & version)
36{
37 // Set up an HTTP request
38 req_.version(version);
39 req_.method(http_method);
40 req_.target(target);
41 req_.set(boost::beast::http::field::host, host);
42 req_.set(boost::beast::http::field::user_agent, BOOST_BEAST_VERSION_STRING);
43
44 // Look up the domain name
45 resolver_.async_resolve(
46 host.c_str(), kPort.c_str(),
47 boost::beast::bind_front_handler(&http_session::on_resolve, shared_from_this()));
48}

50void http_session::on_resolve(
51 boost::beast::error_code ec, boost::asio::ip::tcp::resolver::results_type results)
52{
53 if (ec) return fail(ec, "resolve");
54
55 // Set a timeout on the operation
56 stream_.expires_after(std::chrono::seconds(1));
57
58 // Make the connection on the IP address we get from a lookup
59 stream_.async_connect(
60 results, boost::beast::bind_front_handler(&http_session::on_connect, shared_from_this()));
61}

63void http_session::fail(boost::beast::error_code ec, char const * what)
64{
65 std::cout << "Failure!..." << std::endl;
66 std::cerr << what << ": " << ec.message() << "\n";
67 res_.result(boost::beast::http::status::bad_request);
68 res_.reason() = ec.message();
69 onResponse(res_);
70}

72void http_session::on_connect(
73 boost::beast::error_code ec, boost::asio::ip::tcp::resolver::results_type::endpoint_type)
74{
75 if (ec) return fail(ec, "connect");
76
77 // Set a timeout on the operation
78 stream_.expires_after(std::chrono::seconds(1));
79
80 // Send the HTTP request to the remote host
81 boost::beast::http::async_write(
82 stream_, req_, boost::beast::bind_front_handler(&http_session::on_write, shared_from_this()));
83}

85void http_session::on_write(boost::beast::error_code ec, std::size_t bytes_transferred)
86{
87 boost::ignore_unused(bytes_transferred);
88
89 if (ec) return fail(ec, "write");
90
91 // Receive the HTTP response
92 boost::beast::http::async_read(
93 stream_, buffer_, res_,
94 boost::beast::bind_front_handler(&http_session::on_read, shared_from_this()));
95}

97void http_session::on_read(boost::beast::error_code ec, std::size_t bytes_transferred)
98{
99 boost::ignore_unused(bytes_transferred);
100
101 if (ec) return fail(ec, "read");
102
103 // Gracefully close the socket
104 stream_.socket().shutdown(boost::asio::ip::tcp::socket::shutdown_both, ec);
105
106 // not_connected happens sometimes so don't bother reporting it.
107 if (ec && ec != boost::beast::errc::not_connected) return fail(ec, "shutdown");
108
109 // If we get here then the connection is closed gracefully
110 onResponse(res_);
111}
