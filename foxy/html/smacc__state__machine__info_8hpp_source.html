190 template <typename Ev, typename Dst, typename Tag>
191 void processTransitionAux(
192  smacc2::Transition<Ev, Dst, Tag> *, std::shared_ptr<SmaccStateInfo> & sourceState, bool history,
193  TypeInfo::Ptr & transitionTypeInfo)
194 {
195  RCLCPP_INFO(
196  globalNh_->get_logger(), "State %s Walker transition: %s", sourceState->toShortName().c_str(),
197  demangleSymbol(typeid(Ev).name()).c_str());
198  std::string transitionTag;
199  std::string transitionType;
200 
201  if (typeid(Tag) != typeid(default_transition_name))
202  {
203  transitionTag = demangleSymbol<Tag>();
204  transitionType = getTransitionType<Tag>();
205  RCLCPP_DEBUG_STREAM(globalNh_->get_logger(), "TRANSITION TYPE:" << transitionType);
206  }
207  else
208  {
209  transitionTag = "";
210  automaticTransitionTag<Ev>(transitionTag);
211  automaticTransitionType<Ev>(transitionType);
212  }
213 
214  RCLCPP_INFO_STREAM(globalNh_->get_logger(), "Transition tag: " << transitionTag);
215 
216  if (!sourceState->stateMachine_->containsState<Dst>())
217  {
218  auto realparentState = sourceState->stateMachine_->getState<typename Dst::TContext>();
219  auto siblingnode = sourceState->stateMachine_->createState<Dst>(realparentState);
220 
221  // auto siblingnode = sourceState->stateMachine_->createState<Dst>(sourceState->parentState_);
222  WalkStatesExecutor<Dst>::walkStates(siblingnode, true);
223  sourceState->declareTransition<Ev>(
224  siblingnode, transitionTag, transitionType, history, transitionTypeInfo);
225  }
226  else
227  {
228  // auto realparentState = sourceState->stateMachine_->getState<typename Dst::TContext>();
229  // auto siblingnode = sourceState->stateMachine_->createState<Dst>(realparentState);
230 
231  auto siblingnode = sourceState->stateMachine_->getState<Dst>();
232  sourceState->declareTransition<Ev>(
233  siblingnode, transitionTag, transitionType, history, transitionTypeInfo);
234  }

248  transitionInfo.destinyState = dstState;
249  transitionInfo.transitionTypeInfo = transitionTypeInfo;
250 
251  if (transitionTag != "")
252  transitionInfo.transitionTag = transitionTag;
253  else
254  transitionInfo.transitionTag = "Transition_" + std::to_string(transitionInfo.index);
255 
256  transitionInfo.transitionType = transitionType;
257  transitionInfo.historyNode = history;
258 
259  transitionInfo.eventInfo =
260  std::make_shared<SmaccEventInfo>(transitionTypeInfo->templateParameters.front());
261 
262  EventLabel<EvType>(transitionInfo.eventInfo->label);
263  // RCLCPP_DEBUG_STREAM(getLogger(), "LABEL: " << transitionInfo.eventInfo->label);
264 
265  transitions_.push_back(transitionInfo);

425  std::shared_ptr<SmaccStateInfo> & parentState, bool rootInitialNode)
426 {

428  auto currentname = demangledTypeName<InitialStateType>();
429 
430  std::shared_ptr<SmaccStateInfo> targetState;
431 
432  if (!rootInitialNode)
433  {
434  if (parentState->stateMachine_->containsState<InitialStateType>())
435  {
436  // it already exist: break;
437  return;
438  }
439 
440  targetState = parentState->createChildState<InitialStateType>();
441  }
442  else
443  {
444  targetState = parentState;
445  }
446 
447  CallOnDefinition<InitialStateType>();
448 
449  typedef
450  typename std::remove_pointer<decltype(InitialStateType::smacc_inner_type)>::type InnerType;
451  processSubState<InnerType>(targetState);
452 
453  // -------------------- REACTIONS --------------------
454  typedef typename InitialStateType::reactions reactions;
455  // RCLCPP_INFO_STREAM(getLogger(),"state machine initial state reactions: "
456  // << demangledTypeName<reactions>());
457 
458  processTransitions<reactions>(targetState);
459 }
