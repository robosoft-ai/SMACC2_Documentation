43 void CbAbsoluteRotate::onEntry()
44 {
45  listener = std::make_shared<tf2_ros::Buffer>(getNode()->get_clock());
46  double goal_angle = this->absoluteGoalAngleDegree;
47 
48  RCLCPP_INFO_STREAM(getLogger(), "[CbAbsoluteRotate] Absolute yaw Angle:" << goal_angle);
49 
50  auto plannerSwitcher = this->moveBaseClient_->getComponent<PlannerSwitcher>();
51  // this should work better with a coroutine and await
52  // this->plannerSwitcher_->setForwardPlanner();
53 
54  if (spinningPlanner && *spinningPlanner == SpiningPlanner::PureSpinning)
55  {
56  plannerSwitcher->setPureSpinningPlanner();
57  }
58  else
59  {
60  plannerSwitcher->setDefaultPlanners();
61  }
62 
63  updateTemporalBehaviorParameters(false);
64 
65  auto p = moveBaseClient_->getComponent<cl_nav2z::Pose>();
66  auto referenceFrame = p->getReferenceFrame();
67  auto currentPoseMsg = p->toPoseMsg();
68 
69  ClNav2Z::Goal goal;
70  goal.pose.header.frame_id = referenceFrame;
71  goal.pose.header.stamp = getNode()->now();
72 
73  auto targetAngle = goal_angle * M_PI / 180.0;
74  goal.pose.pose.position = currentPoseMsg.position;
75  tf2::Quaternion q;
76  q.setRPY(0, 0, targetAngle);
77  goal.pose.pose.orientation = tf2::toMsg(q);
78 
79  auto odomTracker_ = moveBaseClient_->getComponent<odom_tracker::OdomTracker>();
80  if (odomTracker_ != nullptr)
81  {
82  odomTracker_->pushPath();
83  odomTracker_->setStartPoint(p->toPoseStampedMsg());
84  odomTracker_->setWorkingMode(odom_tracker::WorkingMode::RECORD_PATH);
85  }
86 
87  auto goalCheckerSwitcher = moveBaseClient_->getComponent<GoalCheckerSwitcher>();
88  goalCheckerSwitcher->setGoalCheckerId("absolute_rotate_goal_checker");
89 
90  RCLCPP_INFO_STREAM(
91  getLogger(),
92  "[CbAbsoluteRotate] current pose yaw: " << tf2::getYaw(currentPoseMsg.orientation));
93  RCLCPP_INFO_STREAM(
94  getLogger(), "[CbAbsoluteRotate] goal pose yaw: " << tf2::getYaw(goal.pose.pose.orientation));
95  moveBaseClient_->sendGoal(goal);
96 }

98 void CbAbsoluteRotate::updateTemporalBehaviorParameters(bool undo)
99 {
100  auto log = this->getLogger();
101 
102  std::string nodename = "/controller_server";
103 
104  auto parameters_client =
105  std::make_shared<rclcpp::AsyncParametersClient>(this->getNode(), nodename);
106 
107  RCLCPP_INFO_STREAM(
108  log, "[CbAbsoluteRotate] using a parameter client to update some controller parameters: "
109  << nodename << ". Waiting service.");
110  parameters_client->wait_for_service();
111 
112  RCLCPP_INFO_STREAM(log, "[CbAbsoluteRotate] Service found: " << nodename << ".");
113 
114  std::string localPlannerName;
115  std::vector<rclcpp::Parameter> parameters;
116 
117  // dynamic_reconfigure::DoubleParameter yaw_goal_tolerance;
118  rclcpp::Parameter yaw_goal_tolerance("goal_checker.yaw_goal_tolerance");
119 
120  rclcpp::Parameter max_vel_theta, min_vel_theta;
121 
122  bool isRosBasePlanner = !spinningPlanner || *spinningPlanner == SpiningPlanner::Default;
123  bool isPureSpinningPlanner = spinningPlanner && *spinningPlanner == SpiningPlanner::PureSpinning;
124 
125  // SELECTING CONTROLLER AND PARAMETERS NAME
126  if (isPureSpinningPlanner)
127  {
128  localPlannerName = "PureSpinningLocalPlanner";
129  max_vel_theta = rclcpp::Parameter(localPlannerName + ".max_angular_z_speed");
130  min_vel_theta = rclcpp::Parameter(localPlannerName + ".min_vel_theta");
131  }
132  else if (isRosBasePlanner)
133  {
134  localPlannerName = "FollowPath";
135  max_vel_theta = rclcpp::Parameter(localPlannerName + ".max_vel_theta");
136  min_vel_theta = rclcpp::Parameter(localPlannerName + ".min_vel_theta");
137  }
138 
139  if (!undo)
140  {
141  if (yawGoalTolerance)
142  {
143  // save old yaw tolerance
144  auto fut = parameters_client->get_parameters(
145  {localPlannerName + ".yaw_goal_tolerance"}, [&](auto futureParameters) {
146  auto params = futureParameters.get();
147  oldYawTolerance = params[0].as_double();
148  });
149 
150  // make synchronous
151  fut.get();
152 
153  yaw_goal_tolerance = rclcpp::Parameter("goal_checker.yaw_goal_tolerance", *yawGoalTolerance);
154  parameters.push_back(yaw_goal_tolerance);
155  RCLCPP_INFO(
156  getLogger(),
157  "[CbAbsoluteRotate] updating yaw tolerance local planner to: %lf, from previous value: "
158  "%lf ",
159  *yawGoalTolerance, this->oldYawTolerance);
160  }
161 
162  if (maxVelTheta)
163  {
164  if (isRosBasePlanner)
165  {
166  // nh.getParam(nodename + "/" + localPlannerName+"/min_vel_theta", oldMinVelTheta);
167  // getCurrentState()->getParam(nodename + "/" + localPlannerName + "/max_vel_theta", oldMaxVelTheta);
168 
169  // save old yaw tolerance
170  auto fut = parameters_client->get_parameters(
171  {localPlannerName + ".max_vel_theta"}, [&](auto futureParameters) {
172  auto params = futureParameters.get();
173  oldMaxVelTheta = params[0].as_double();
174  });
175 
176  // make synchronous
177  fut.get();
178  }
179 
180  max_vel_theta = rclcpp::Parameter(localPlannerName + ".max_vel_theta", *maxVelTheta);
181  min_vel_theta = rclcpp::Parameter(localPlannerName + ".min_vel_theta", -*maxVelTheta);
182  parameters.push_back(max_vel_theta);
183  parameters.push_back(min_vel_theta);
184 
185  RCLCPP_INFO(
186  log,
187  "[CbAbsoluteRotate] updating max vel theta local planner to: %lf, from previous value: "
188  "%lf ",
189  *maxVelTheta, this->oldMaxVelTheta);
190  RCLCPP_INFO(
191  log,
192  "[CbAbsoluteRotate] updating min vel theta local planner to: %lf, from previous value: "
193  "%lf ",
194  -*maxVelTheta, this->oldMinVelTheta);
195  }
196  }
197  else
198  {
199  if (yawGoalTolerance)
200  {
201  yaw_goal_tolerance = rclcpp::Parameter("goal_checker.yaw_goal_tolerance", oldYawTolerance);
202  RCLCPP_INFO(
203  log,
204  "[CbAbsoluteRotate] restoring yaw tolerance local planner from: %lf, to previous value: "
205  "%lf ",
206  *yawGoalTolerance, this->oldYawTolerance);
207  }
208 
209  if (maxVelTheta)
210  {
211  if (isRosBasePlanner)
212  {
213  max_vel_theta = rclcpp::Parameter(localPlannerName + ".max_vel_theta", oldMaxVelTheta);
214  min_vel_theta = rclcpp::Parameter(localPlannerName + ".max_vel_theta", oldMinVelTheta);
215  }
216 
217  parameters.push_back(max_vel_theta);
218  parameters.push_back(min_vel_theta);
219  RCLCPP_INFO(
220  log,
221  "[CbAbsoluteRotate] restoring max vel theta local planner from: %lf, to previous value: "
222  "%lf ",
223  *maxVelTheta, this->oldMaxVelTheta);
224  RCLCPP_INFO(
225  log,
226  "[CbAbsoluteRotate] restoring min vel theta local planner from: %lf, to previous value: "
227  "%lf ",
228  -(*maxVelTheta), this->oldMinVelTheta);
229  }
230  }
231 
232  if (parameters.size())
233  {
234  std::stringstream ss;
235  ss << "Executing asynchronous request. Parameters to update: " << std::endl;
236  for (auto & p : parameters)
237  {
238  ss << p.get_name() << " - " << p.value_to_string() << std::endl;
239  }
240 
241  RCLCPP_INFO_STREAM(getLogger(), "[CbAbsoluteRotate] " << ss.str());
242 
243  auto futureResults = parameters_client->set_parameters(parameters);
244 
245  int i = 0;
246  for (auto & res : futureResults.get())
247  {
248  RCLCPP_INFO_STREAM(
249  getLogger(), "[CbAbsoluteRotate] parameter result: " << parameters[i].get_name() << "="
250  << parameters[i].as_string()
251  << ". Result: " << res.successful);
252  i++;
253  }
254 
255  RCLCPP_INFO(log, "[CbAbsoluteRotate] parameters updated");
256  }
257  else
258  {
259  RCLCPP_INFO(log, "[CbAbsoluteRotate] skipping parameters update");
260  }
261 }
