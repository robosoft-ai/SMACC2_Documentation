158 void OdomTracker::popPath(int popCount, bool keepPreviousPath)
159 {
160  RCLCPP_INFO(getLogger(), "odom_tracker m_mutex acquire");
161  std::lock_guard<std::mutex> lock(m_mutex_);
162 
163  RCLCPP_INFO(getLogger(), "POP PATH ENTRY");
164  this->logStateString();
165 
166  if (!keepPreviousPath)
167  {
168  baseTrajectory_.poses.clear();
169  }
170 
171  while (popCount > 0 && !pathStack_.empty())
172  {
173  auto & stacked = pathStack_.back().poses;
174  baseTrajectory_.poses.insert(baseTrajectory_.poses.begin(), stacked.begin(), stacked.end());
175  pathStack_.pop_back();
176  popCount--;
177 
178  RCLCPP_INFO(getLogger(), "POP PATH Iteration ");
179  this->logStateString();
180  }
181 
182  RCLCPP_INFO(getLogger(), "POP PATH EXITING");
183  this->logStateString();
184  RCLCPP_INFO(getLogger(), "odom_tracker m_mutex release");
185  this->updateAggregatedStackPath();
186 }

285 bool OdomTracker::updateClearPath(const nav_msgs::msg::Odometry & odom)
286 {
287  // we initially accept any message if the queue is empty
289  geometry_msgs::msg::PoseStamped base_pose;
290 
291  base_pose.pose = odom.pose.pose;
292  base_pose.header = odom.header;
293  baseTrajectory_.header = odom.header;
294 
295  bool acceptBackward = false;
296  bool clearingError = false;
297  bool finished = false;
298 
299  while (!finished)
300  {
301  if (
302  baseTrajectory_.poses.size() <=
303  1) // we at least keep always the first point of the forward path when clearing
304  // (this is important for backwards planner replanning and not losing the
305  // last goal)
306  {
307  acceptBackward = false;
308  finished = true;
309  }
310  else
311  {
312  auto & carrotPose = baseTrajectory_.poses.back().pose;
313  auto & carrotPoint = carrotPose.position;
314 
315  double carrotAngle = tf2::getYaw(carrotPose.orientation);
316 
317  auto & currePose = base_pose.pose;
318  auto & currePoint = currePose.position;
319  double currentAngle = tf2::getYaw(currePose.orientation);
320 
321  double lastpointdist = p2pDistance(carrotPoint, currePoint);
322  double goalAngleOffset = fabs(angles::shortest_angular_distance(carrotAngle, currentAngle));
323 
324  acceptBackward = !baseTrajectory_.poses.empty() &&
325  lastpointdist < clearPointDistanceThreshold_ &&
326  goalAngleOffset < clearAngularDistanceThreshold_;
327 
328  clearingError = lastpointdist > 2 * clearPointDistanceThreshold_;
329  RCLCPP_DEBUG_STREAM(
330  getLogger(), "[OdomTracker] clearing (accepted: " << acceptBackward
331  << ") linerr: " << lastpointdist
332  << ", anglerr: " << goalAngleOffset);
333  }
334 
335  // RCLCPP_INFO(getLogger(),"Backwards, last distance: %lf < %lf accept: %d", dist,
336  // minPointDistanceBackwardThresh_, acceptBackward);
337  if (
338  acceptBackward &&
339  baseTrajectory_.poses.size() > 1) /* we always leave at least one item, specially interesting
340  for the backward local planner reach the backwards goal
341  with enough precision*/
342  {
343  baseTrajectory_.poses.pop_back();
344  }
345  else if (clearingError)
346  {
347  finished = true;
348  RCLCPP_WARN(getLogger(), "[OdomTracker] Incorrect odom clearing motion.");
349  }
350  else
351  {
352  finished = true;
354  }
355  }
356 
357  return acceptBackward;
358 }
364 bool OdomTracker::updateRecordPath(const nav_msgs::msg::Odometry & odom)
365 {
367  geometry_msgs::msg::PoseStamped base_pose;
368 
369  base_pose.pose = odom.pose.pose;
370  base_pose.header = odom.header;
371  baseTrajectory_.header = odom.header;
372 
373  bool enqueueOdomMessage = false;
374 
375  double dist = -1;
376  if (baseTrajectory_.poses.empty())
377  {
378  enqueueOdomMessage = true;
379  }
380  else
381  {
382  const auto & prevPose = baseTrajectory_.poses.back().pose;
383  const geometry_msgs::msg::Point & prevPoint = prevPose.position;
384  double prevAngle = tf2::getYaw(prevPose.orientation);
385 
386  const geometry_msgs::msg::Point & currePoint = base_pose.pose.position;
387  double currentAngle = tf2::getYaw(base_pose.pose.orientation);
388 
389  dist = p2pDistance(prevPoint, currePoint);
390  double goalAngleOffset = fabs(angles::shortest_angular_distance(prevAngle, currentAngle));
391 
392  // RCLCPP_WARN(getLogger(),"dist %lf vs min %lf", dist, recordPointDistanceThreshold_);
393 
394  if (dist > recordPointDistanceThreshold_ || goalAngleOffset > recordAngularDistanceThreshold_)
395  {
396  enqueueOdomMessage = true;
397  }
398  else
399  {
400  // RCLCPP_WARN(getLogger(),"skip odom, dist: %lf", dist);
401  enqueueOdomMessage = false;
402  }
403  }
404 
405  if (enqueueOdomMessage)
406  {
407  baseTrajectory_.poses.push_back(base_pose);
408  }
409 
410  return enqueueOdomMessage;
411 }
