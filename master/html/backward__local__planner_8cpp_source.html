244 void BackwardLocalPlanner::computeCurrentEuclideanAndAngularErrorsToCarrotGoal(
245  const geometry_msgs::msg::PoseStamped & tfpose, double & dist, double & angular_error)
246 {
247  double angle = tf2::getYaw(tfpose.pose.orientation);
248  auto & carrot_pose = backwardsPlanPath_[currentCarrotPoseIndex_];
249  const geometry_msgs::msg::Point & carrot_point = carrot_pose.pose.position;
250 
251  tf2::Quaternion carrot_orientation;
252  tf2::fromMsg(carrot_pose.pose.orientation, carrot_orientation);
253  geometry_msgs::msg::Pose currentPoseDebugMsg = tfpose.pose;
254 
255  // take error from the current position to the path point
256  double dx = carrot_point.x - tfpose.pose.position.x;
257  double dy = carrot_point.y - tfpose.pose.position.y;
258 
259  dist = sqrt(dx * dx + dy * dy);
260 
261  double pangle = tf2::getYaw(carrot_orientation);
262  angular_error = fabs(angles::shortest_angular_distance(pangle, angle));
263 
264  RCLCPP_INFO_STREAM(
265  nh_->get_logger(), "[BackwardLocalPlanner] Compute carrot errors from current pose. (linear "
266  << dist << ")(angular " << angular_error << ")" << std::endl
267  << "Current carrot pose: " << std::endl
268  << carrot_pose << std::endl
269  << "Current actual pose:" << std::endl
270  << currentPoseDebugMsg);
271 }

278 bool BackwardLocalPlanner::updateCarrotGoal(const geometry_msgs::msg::PoseStamped & tfpose)
279 {
280  RCLCPP_INFO_STREAM(nh_->get_logger(), "[BackwardsLocalPlanner] --- Carrot update ---");
281  double disterr = 0, angleerr = 0;
282  // iterate the point from the current position and backward until reaching a new goal point in the path
283  // this algorithm among other advantages has that skip the looping with an eager global planner
284  // that recalls the same plan (the already performed part of the plan in the current pose is skipped)
285  while (currentCarrotPoseIndex_ < (long)backwardsPlanPath_.size() - 1)
286  {
287  computeCurrentEuclideanAndAngularErrorsToCarrotGoal(tfpose, disterr, angleerr);
288 
289  RCLCPP_INFO_STREAM(
290  nh_->get_logger(), "[BackwardsLocalPlanner] update carrot goal: Current index: "
291  << currentCarrotPoseIndex_ << "/" << backwardsPlanPath_.size());
292  RCLCPP_INFO(
293  nh_->get_logger(),
294  "[BackwardsLocalPlanner] update carrot goal: linear error %lf, angular error: %lf", disterr,
295  angleerr);
296 
297  // target pose found, goal carrot tries to escape!
298  if (disterr < carrot_distance_ && angleerr < carrot_angular_distance_)
299  {
300  currentCarrotPoseIndex_++;
301  resetDivergenceDetection();
302  RCLCPP_INFO_STREAM(
303  nh_->get_logger(), "[BackwardsLocalPlanner] move carrot fw "
304  << currentCarrotPoseIndex_ << "/" << backwardsPlanPath_.size());
305  }
306  else
307  {
308  // carrot already escaped
309  break;
310  }
311  }
312  // RCLCPP_INFO(nh_->get_logger(),"[BackwardsLocalPlanner] computing angular error");
313  if (
314  currentCarrotPoseIndex_ >= (long)backwardsPlanPath_.size() - 1 && backwardsPlanPath_.size() > 0)
315  {
316  currentCarrotPoseIndex_ = backwardsPlanPath_.size() - 1;
317  // reupdated errors
318  computeCurrentEuclideanAndAngularErrorsToCarrotGoal(tfpose, disterr, angleerr);
319  }
320 
321  RCLCPP_INFO(
322  nh_->get_logger(), "[BackwardsLocalPlanner] Current index carrot goal: %d",
323  currentCarrotPoseIndex_);
324  RCLCPP_INFO(
325  nh_->get_logger(),
326  "[BackwardsLocalPlanner] Update carrot goal: linear error %lf (xytol: %lf), angular error: "
327  "%lf",
328  disterr, xy_goal_tolerance_, angleerr);
329 
330  bool carrotInGoalLinearRange = disterr < xy_goal_tolerance_;
331  RCLCPP_INFO(
332  nh_->get_logger(), "[BackwardsLocalPlanner] carrot in goal radius: %d",
333  carrotInGoalLinearRange);
334 
335  RCLCPP_INFO(nh_->get_logger(), "[BackwardsLocalPlanner] ---End carrot update---");
336 
337  return carrotInGoalLinearRange;
338 }

347 bool BackwardLocalPlanner::divergenceDetectionUpdate(const geometry_msgs::msg::PoseStamped & tfpose)
348 {
349  double disterr = 0, angleerr = 0;
350  computeCurrentEuclideanAndAngularErrorsToCarrotGoal(tfpose, disterr, angleerr);
351 
352  RCLCPP_INFO_STREAM(
353  nh_->get_logger(), "[BackwardLocalPlanner] Divergence check. carrot goal distance. was: "
354  << divergenceDetectionLastCarrotLinearDistance_
355  << ", now it is: " << disterr);
356  if (disterr > divergenceDetectionLastCarrotLinearDistance_)
357  {
358  // candidate of divergence, we do not throw the divergence alarm yet
359  // but we neither update the distance since it is worse than the one
360  // we had previously with the same carrot.
361  const double MARGIN_FACTOR = 1.2;
362  if (disterr > MARGIN_FACTOR * divergenceDetectionLastCarrotLinearDistance_)
363  {
364  RCLCPP_ERROR_STREAM(
365  nh_->get_logger(),
366  "[BackwardLocalPlanner] Divergence detected. The same carrot goal distance was previously: "
367  << divergenceDetectionLastCarrotLinearDistance_ << "but now it is: " << disterr);
368  return true;
369  }
370  else
371  {
372  // divergence candidate
373  return false;
374  }
375  }
376  else
377  {
378  // update:
379  divergenceDetectionLastCarrotLinearDistance_ = disterr;
380  return false;
381  }

384 bool BackwardLocalPlanner::checkCarrotHalfPlainConstraint(
385  const geometry_msgs::msg::PoseStamped & tfpose)
386 {
387  // this function is specially useful when we want to reach the goal with a lot
388  // of precision. We may pass the goal and then the controller enters in some
389  // unstable state. With this, we are able to detect when stop moving.
390 
391  // only apply if the carrot is in goal position and also if we are not in a pure spinning behavior v!=0
392 
393  auto & carrot_pose = backwardsPlanPath_[currentCarrotPoseIndex_];
394  const geometry_msgs::msg::Point & carrot_point = carrot_pose.pose.position;
395  double yaw = tf2::getYaw(carrot_pose.pose.orientation);
396 
397  // direction vector
398  double vx = cos(yaw);
399  double vy = sin(yaw);
400 
401  // line implicit equation
402  // ax + by + c = 0
403  double c = -vx * carrot_point.x - vy * carrot_point.y;
404  const double C_OFFSET_METERS = 0.05; // 5 cm
405  double check = vx * tfpose.pose.position.x + vy * tfpose.pose.position.y + c + C_OFFSET_METERS;
406 
407  RCLCPP_INFO_STREAM(
408  nh_->get_logger(),
409  "[BackwardLocalPlanner] half plane constraint:" << vx << "*" << carrot_point.x << " + " << vy
410  << "*" << carrot_point.y << " + " << c);
411  RCLCPP_INFO_STREAM(
412  nh_->get_logger(), "[BackwardLocalPlanner] constraint evaluation: "
413  << vx << "*" << tfpose.pose.position.x << " + " << vy << "*"
414  << tfpose.pose.position.y << " + " << c << " = " << check);
415 
416  return check < 0;
417 }

507  }
508  else
509  {
510  RCLCPP_INFO_STREAM(
511  nh_->get_logger(), "[BackwardLocalPlanner] could not get tolerances from goal checker");
512  }
513  }
514 
515  RCLCPP_INFO(
516  nh_->get_logger(),
517  "[BackwardLocalPlanner] ------------------- LOCAL PLANNER LOOP -----------------");
518 
519  geometry_msgs::msg::TwistStamped cmd_vel;
520  RCLCPP_INFO(nh_->get_logger(), "[BackwardLocalPlanner] LOCAL PLANNER LOOP");
521  geometry_msgs::msg::PoseStamped paux;
522  geometry_msgs::msg::PoseStamped tfpose;
523 
524  if (!costmapRos_->getRobotPose(tfpose))
525  {
526  RCLCPP_ERROR(

528  "[BackwardLocalPlanner] missing robot pose, canceling compute Velocity Command");
529  } // it is not working in the pure spinning reel example, maybe the hyperplane check is enough
530  bool divergenceDetected = false;
531 
532  bool emergency_stop = false;
533  if (divergenceDetected)
534  {
535  RCLCPP_ERROR(
536  nh_->get_logger(), "[BackwardLocalPlanner] Divergence detected. Sending emergency stop.");
537  emergency_stop = true;
538  }
539 
540  bool carrotInLinearGoalRange = updateCarrotGoal(tfpose);
541  RCLCPP_INFO_STREAM(nh_->get_logger(), "[BackwardLocalPlanner] carrot goal created");
542 
543  if (emergency_stop)
544  {
545  cmd_vel.twist.linear.x = 0;
546  cmd_vel.twist.angular.z = 0;
547  RCLCPP_INFO_STREAM(
548  nh_->get_logger(), "[BackwardLocalPlanner] emergency stop, exit compute commands");
549  // return false;
550  return cmd_vel;
551  }
552 
553  // ------ Evaluate the current context ----
554  double rho_error, betta_error, alpha_error;
555 
556  // getting carrot goal information
557  tf2::Quaternion q;
558  tf2::fromMsg(tfpose.pose.orientation, q);
559 
560  RCLCPP_INFO_STREAM(
561  nh_->get_logger(), "[BackwardLocalPlanner] carrot goal: " << currentCarrotPoseIndex_ << "/"
562  << backwardsPlanPath_.size());
563  const geometry_msgs::msg::PoseStamped & carrotgoalpose =
564  backwardsPlanPath_[currentCarrotPoseIndex_];
565  RCLCPP_INFO_STREAM(
566  nh_->get_logger(), "[BackwardLocalPlanner] carrot goal pose current index: "
567  << currentCarrotPoseIndex_ << "/" << backwardsPlanPath_.size() << ": "
568  << carrotgoalpose);
569  const geometry_msgs::msg::Point & carrotGoalPosition = carrotgoalpose.pose.position;
570 
571  tf2::Quaternion goalQ;
572  tf2::fromMsg(carrotgoalpose.pose.orientation, goalQ);
573  RCLCPP_INFO_STREAM(nh_->get_logger(), "[BackwardLocalPlanner] -- Control Policy --");
574  // goal orientation (global frame)
575  double betta = tf2::getYaw(goalQ);
576  RCLCPP_INFO_STREAM(nh_->get_logger(), "[BackwardLocalPlanner] goal orientation: " << betta);
577  betta = betta + betta_offset_;

580  double dy = carrotGoalPosition.y - tfpose.pose.position.y;
581

583  rho_error = sqrt(dx * dx + dy * dy);
584 
585  // heading to goal angle
586  double theta = tf2::getYaw(q);
587  double alpha = atan2(dy, dx);
588  alpha = alpha + alpha_offset_;
589 
590  alpha_error = angles::shortest_angular_distance(alpha, theta);
591  betta_error = angles::shortest_angular_distance(betta, theta);
592  //------------- END CONTEXT EVAL ----------
593

595  bool currentPoseInGoal =
596  checkCurrentPoseInGoalRange(tfpose, velocity, betta_error, linearGoalReached, goal_checker);
597 
598  // Make sure the robot is very close to the goal and it is really in the the last goal point.
599  bool carrotInFinalGoalIndex = currentCarrotPoseIndex_ == (int)backwardsPlanPath_.size() - 1;
600 
601  // checking if we are really in the end goal pose
602  if (currentPoseInGoal && carrotInFinalGoalIndex)
603  {
604  goalReached_ = true;
605  // backwardsPlanPath_.clear();
606  RCLCPP_INFO_STREAM(
607  nh_->get_logger(),
608  "[BackwardLocalPlanner] GOAL REACHED. Send stop command and skipping trajectory collision: "
609  << cmd_vel.twist);
610  cmd_vel.twist.linear.x = 0;
611  cmd_vel.twist.angular.z = 0;
612  return cmd_vel;
613  }
614  else if (
615  carrotInLinearGoalRange &&
616  linearGoalReached) // checking if we are in the end goal point but with incorrect
617  // orientation
618  {
619  // this means that we are not in the final angular distance, and we may even not be in the last carrot index
620  // (several intermediate angular poses until the last goal pose)
621  inGoalPureSpinningState_ = true;
622  }
623 
624  // --------------------
625  double vetta, gamma;
626  if (straightBackwardsAndPureSpinningMode_)
627  {
628  // decorated control rule for this mode
629  this->straightBackwardsAndPureSpinCmd(
630  tfpose, vetta, gamma, alpha_error, betta_error, rho_error);
631  }
632  else // default free navigation backward motion mode
633  {
634  // regular control rule
635  vetta = k_rho_ * rho_error;
636  gamma = k_alpha_ * alpha_error + k_betta_ * betta_error;
637 
638  // Even if we are in free navigation, we can enter in the pure spinning state.
639  // then, the linear motion is deactivated.
640  if (inGoalPureSpinningState_)
641  {
642  RCLCPP_INFO(
643  nh_->get_logger(),
644  "[BackwardLocalPlanner] we entered in a pure spinning state even in not pure-spining "
645  "configuration, "
646  "carrotDistanceGoalReached: %d",
647  carrotInLinearGoalRange);
648  gamma = k_betta_ * betta_error;
649  vetta = 0;
650  }
651 
652  // classical control to reach a goal backwards
653  }
654 
655  // Apply command and Clamp to limits
656  cmd_vel.twist.linear.x = vetta;
657  cmd_vel.twist.angular.z = gamma;
658 
659  if (cmd_vel.twist.linear.x > max_linear_x_speed_)
660  {

662  }

684  << " carrot in linear goal range: " << carrotInLinearGoalRange << std::endl

928 bool BackwardLocalPlanner::resamplePrecisePlan()
929 {
930  // this algorithm is really important to have a precise carrot (linear or angular)
931  // and not being considered as a divergence from the path
932 
933  RCLCPP_INFO(nh_->get_logger(), "[BackwardLocalPlanner] resample precise");
934  if (backwardsPlanPath_.size() <= 1)
935  {
936  RCLCPP_INFO_STREAM(
937  nh_->get_logger(),
938  "[BackwardLocalPlanner] resample precise skipping, size: " << backwardsPlanPath_.size());
939  return false;
940  }
941 
942  int counter = 0;
943  double maxallowedAngularError = 0.45 * this->carrot_angular_distance_; // nyquist
944  double maxallowedLinearError = 0.45 * this->carrot_distance_; // nyquist
945 
946  for (int i = 0; i < (int)backwardsPlanPath_.size() - 1; i++)
947  {
948  RCLCPP_INFO_STREAM(nh_->get_logger(), "[BackwardLocalPlanner] resample precise, check: " << i);
949  auto & currpose = backwardsPlanPath_[i];
950  auto & nextpose = backwardsPlanPath_[i + 1];
951 
952  tf2::Quaternion qCurrent, qNext;
953  tf2::fromMsg(currpose.pose.orientation, qCurrent);
954  tf2::fromMsg(nextpose.pose.orientation, qNext);
955 
956  double dx = nextpose.pose.position.x - currpose.pose.position.x;
957  double dy = nextpose.pose.position.y - currpose.pose.position.y;
958  double dist = sqrt(dx * dx + dy * dy);
959 
960  bool resample = false;
961  if (dist > maxallowedLinearError)
962  {
963  RCLCPP_INFO_STREAM(
964  nh_->get_logger(), "[BackwardLocalPlanner] resampling point, linear distance:"
965  << dist << "(" << maxallowedLinearError << ")" << i);
966  resample = true;
967  }
968  else
969  {
970  double currentAngle = tf2::getYaw(qCurrent);
971  double nextAngle = tf2::getYaw(qNext);
972 
973  double angularError = fabs(angles::shortest_angular_distance(currentAngle, nextAngle));
974  if (angularError > maxallowedAngularError)
975  {
976  resample = true;
977  RCLCPP_INFO_STREAM(
978  nh_->get_logger(), "[BackwardLocalPlanner] resampling point, angular distance:"
979  << angularError << "(" << maxallowedAngularError << ")" << i);
980  }
981  }
982 
983  if (resample)
984  {
985  geometry_msgs::msg::PoseStamped pintermediate;
986  auto duration = rclcpp::Time(nextpose.header.stamp) - rclcpp::Time(currpose.header.stamp);
987 
988  pintermediate.header.frame_id = currpose.header.frame_id;
989  pintermediate.header.stamp = rclcpp::Time(currpose.header.stamp) + duration * 0.5;
990 
991  pintermediate.pose.position.x = 0.5 * (currpose.pose.position.x + nextpose.pose.position.x);
992  pintermediate.pose.position.y = 0.5 * (currpose.pose.position.y + nextpose.pose.position.y);
993  pintermediate.pose.position.z = 0.5 * (currpose.pose.position.z + nextpose.pose.position.z);
994  tf2::Quaternion intermediateQuat = tf2::slerp(qCurrent, qNext, 0.5);
995  pintermediate.pose.orientation = tf2::toMsg(intermediateQuat);
996 
997  this->backwardsPlanPath_.insert(this->backwardsPlanPath_.begin() + i + 1, pintermediate);
998 
999  // retry this point
1000  i--;
1001  counter++;
1002  }
1003  }
1004 
1005  RCLCPP_INFO_STREAM(
1006  nh_->get_logger(), "[BackwardLocalPlanner] End resampling. resampled:" << counter
1007  << " new inserted poses "
1008  "during precise "
1009  "resmapling.");
1010  return true;
1011 }
