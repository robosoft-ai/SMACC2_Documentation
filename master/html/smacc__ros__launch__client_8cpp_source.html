42 std::future<std::string> ClRosLaunch::executeRosLaunch(
43  std::string packageName, std::string launchFileName, std::function<bool()> cancelCondition)
44 {
45  return std::async(std::launch::async, [=]() {
46  RCLCPP_WARN_STREAM(
47  rclcpp::get_logger("smacc2"), "[ClRosLaunch] starting ros launch thread " << ss.str());
48  std::stringstream cmd;
49  cmd << "ros2 launch " << packageName << " " << launchFileName;
50 
51  std::array<char, 128> buffer;
52  std::string result;
53  std::unique_ptr<FILE, decltype(&pclose)> pipe(popen(cmd.str().c_str(), "r"), pclose);
54  if (!pipe)
55  {
56  throw std::runtime_error("popen() failed!");
57  }
58 
59  std::stringstream ss;
60  bool cancelled = false;
61  while (fgets(buffer.data(), buffer.size(), pipe.get()) != nullptr &&
62  !(cancelled = cancelCondition()))
63  {
64  ss << buffer.data();
65  }
66 
67  result = ss.str();
68  RCLCPP_WARN_STREAM(rclcpp::get_logger("smacc2"), "[ClRosLaunch] RESULT = \n " << ss.str());
69  return result;
70  });
71 }
