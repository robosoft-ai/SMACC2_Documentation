43 void CbWaitNav2Nodes::onMessageReceived(const bond::msg::Status & msg)
44 {
45  auto value = fromString(msg.id);
46  bool updated = false;
47  // RCLCPP_INFO(getLogger(), "[CbWaitNav2Nodes] received '%s'", msg.id.c_str());
48 
49  if (receivedAliveMsg_.count(value) && !receivedAliveMsg_[value])
50  {
51  RCLCPP_INFO(getLogger(), "[CbWaitNav2Nodes] '%s' alive received", msg.id.c_str());
52  receivedAliveMsg_[value] = true;
53  updated = true;
54  }
55 
56  if (updated)
57  {
58  bool success = true;
59  std::stringstream ss;
60  for (auto & pair : receivedAliveMsg_)
61  {
62  if (!pair.second)
63  {
64  success = false;
65  }
66  ss << "[CbWaitNav2Nodes] - " << toString(pair.first) << ":" << pair.second << std::endl;
67  }
68 
69  RCLCPP_INFO(getLogger(), ss.str().c_str());
70 
71  if (success)
72  {
73  RCLCPP_INFO(getLogger(), "[CbWaitNav2Nodes] success event");
74  this->postSuccessEvent();
75  sub_ = nullptr;
76  }
77  else
78  {
79  RCLCPP_INFO(getLogger(), "[CbWaitNav2Nodes] still missing nodes");
80  }
81  }
