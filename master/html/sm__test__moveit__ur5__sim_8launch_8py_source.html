31def construct_angle_radians(loader, node):
32 """utility function to construct radian values from yaml."""
33 value = loader.construct_scalar(node)
34 try:
35 return float(value)
36 except SyntaxError:
37 raise Exception("invalid expression: %s" % value)
38
39
40def construct_angle_degrees(loader, node):
41 """utility function for converting degrees into radians from yaml."""
42 return math.radians(construct_angle_radians(loader, node))

45def load_yaml(package_name, file_path):
46 package_path = get_package_share_directory(package_name)
47 absolute_file_path = os.path.join(package_path, file_path)
48
49 try:
50 yaml.SafeLoader.add_constructor("!radians", construct_angle_radians)
51 yaml.SafeLoader.add_constructor("!degrees", construct_angle_degrees)
52 except Exception:
53 raise Exception("yaml support not available; install python-yaml")
54
55 try:
56 with open(absolute_file_path) as file:
57 return yaml.safe_load(file)
58 except OSError: # parent of IOError, OSError *and* WindowsError where available
59 return None
