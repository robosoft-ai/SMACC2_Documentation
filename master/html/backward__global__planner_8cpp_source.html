137  geometry_msgs::msg::Point start, end;
138  start.x = pose.position.x;
139  start.y = pose.position.y;
140 
141  end.x = pose.position.x + 0.5 * cos(phi);

144  marker.points.push_back(start);

147  visualization_msgs::msg::MarkerArray ma;
148  ma.markers.push_back(marker);

158 void BackwardGlobalPlanner::createDefaultBackwardPath(
159  const geometry_msgs::msg::PoseStamped & start, const geometry_msgs::msg::PoseStamped & goal,
160  std::vector<geometry_msgs::msg::PoseStamped> & plan)
161 {
162  auto q = start.pose.orientation;
163 
164  geometry_msgs::msg::PoseStamped pose;
165  pose = start;
166 
167  double dx = start.pose.position.x - goal.pose.position.x;
168  double dy = start.pose.position.y - goal.pose.position.y;
169 
170  double length = sqrt(dx * dx + dy * dy);
171 
172  geometry_msgs::msg::PoseStamped prevState;
173  if (length > skip_straight_motion_distance_)
174  {
175  // skip initial pure spinning and initial straight motion
176  // RCLCPP_INFO(nh_->get_logger(), "1 - heading to goal position pure spinning");
177  double heading_direction = atan2(dy, dx);
178  double startyaw = tf2::getYaw(q);
179  double offset = angles::shortest_angular_distance(startyaw, heading_direction);
180  heading_direction = startyaw + offset;
181 
182  prevState = cl_move_base_z::makePureSpinningSubPlan(
183  start, heading_direction, plan, puresSpinningRadStep_);
184  // RCLCPP_INFO(nh_->get_logger(), "2 - going forward keep orientation pure straight");
185 
186  prevState =
187  cl_move_base_z::makePureStraightSubPlan(prevState, goal.pose.position, length, plan);
188  }
189  else
190  {
191  prevState = start;
192  }
193 
194  RCLCPP_WARN_STREAM(
195  nh_->get_logger(), "[BackwardGlobalPlanner] backward global plan size: " << plan.size());
196 }

203 nav_msgs::msg::Path BackwardGlobalPlanner::createPlan(
204  const geometry_msgs::msg::PoseStamped & start, const geometry_msgs::msg::PoseStamped & goal)
205 {
206  RCLCPP_INFO_STREAM(
207  nh_->get_logger(), "[BackwardGlobalPlanner] goal frame id: "
208  << goal.header.frame_id << " pose: " << goal.pose.position);
209  RCLCPP_INFO_STREAM(
210  nh_->get_logger(), "[BackwardGlobalPlanner] goal pose frame id: " << goal.header.frame_id);
211 
212  rclcpp::Duration ttol = rclcpp::Duration::from_seconds(transform_tolerance_);
213  //---------------------------------------------------------------------
214  geometry_msgs::msg::PoseStamped transformedStart;
215  nav_2d_utils::transformPose(tf_, costmap_ros_->getGlobalFrameID(), start, transformedStart, ttol);
216  transformedStart.header.frame_id = costmap_ros_->getGlobalFrameID();
217  //---------------------------------------------------------------------
218  geometry_msgs::msg::PoseStamped transformedGoal;
219  nav_2d_utils::transformPose(tf_, costmap_ros_->getGlobalFrameID(), goal, transformedGoal, ttol);
220  transformedGoal.header.frame_id = costmap_ros_->getGlobalFrameID();
221 
222  //---------------------------------------------------------------------
223  std::vector<geometry_msgs::msg::PoseStamped> plan;
224  this->createDefaultBackwardPath(transformedStart, transformedGoal, plan);
225 
226  publishGoalMarker(transformedGoal.pose, 1.0, 0, 1.0);
227 
228  nav_msgs::msg::Path planMsg;
229  planMsg.poses = plan;
230  planMsg.header.frame_id = this->costmap_ros_->getGlobalFrameID();
231 
232  //---------------------------------------------------------------------
233  // check plan rejection if obstacle is found
234  bool acceptedGlobalPlan = true;
235 
236  auto costmap2d = this->costmap_ros_->getCostmap();
237  for (auto & p : plan)
238  {
239  unsigned int mx, my;
240  costmap2d->worldToMap(p.pose.position.x, p.pose.position.y, mx, my);
241  auto cost = costmap2d->getCost(mx, my);
242 
243  // static const unsigned char NO_INFORMATION = 255;
244  // static const unsigned char LETHAL_OBSTACLE = 254;
245  // static const unsigned char INSCRIBED_INFLATED_OBSTACLE = 253;
246  // static const unsigned char FREE_SPACE = 0;
247  if (cost >= nav2_costmap_2d::INSCRIBED_INFLATED_OBSTACLE)
248  {
249  acceptedGlobalPlan = false;
250  break;
251  }
252  }
253 
254  if (acceptedGlobalPlan)
255  {
256  RCLCPP_WARN_STREAM(
257  nh_->get_logger(),
258  "[BackwardGlobalPlanner] backward plan request is not accepted, returning "
259  "empty path");
260  planMsg.poses.clear();
261  }
262 
263  RCLCPP_WARN_STREAM(
264  nh_->get_logger(), "[BackwardGlobalPlanner] backward global plan publishing path. poses count: "
265  << planMsg.poses.size());
266  planPub_->publish(planMsg);
267 
268  return planMsg;
269 }
