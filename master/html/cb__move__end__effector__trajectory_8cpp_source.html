87  for (size_t k = 0; k < this->endEffectorTrajectory_.size(); k++)

102 
103  RCLCPP_INFO_STREAM(
104  getLogger(), "[ComputeJointTrajectoryErrorCode] IK request: " << k << " " << *req);
105 
106  auto resfut = iksrv_->async_send_request(req);
107 
108  auto status = resfut.wait_for(3s);
109  if (status == std::future_status::ready)
110  {
111  //if (rclcpp::spin_until_future_complete(getNode(), resfut) == rclcpp::FutureReturnCode::SUCCESS)
112  //{
113  auto & prevtrajpoint = trajectory.back();
114  //jointPositions.clear();
115 
116  auto res = resfut.get();
117  std::stringstream ss;
118  for (size_t j = 0; j < res->solution.joint_state.position.size(); j++)
119  {
120  auto & jointname = res->solution.joint_state.name[j];
121  auto it = std::find(currentjointnames.begin(), currentjointnames.end(), jointname);
122  if (it != currentjointnames.end())
123  {
124  int index = std::distance(currentjointnames.begin(), it);
125  jointPositions[index] = res->solution.joint_state.position[j];
126  ss << jointname << "(" << index << "): " << jointPositions[index] << std::endl;
127  }
128  }
129 
130  // continuity check
131  size_t jointindex = 0;
132  int discontinuityJointIndex = -1;
133  double discontinuityDeltaJointIndex = -1;
134  double deltajoint;
135 
136  bool check = k > 0 || !allowInitialTrajectoryStateJointDiscontinuity_ ||
137  allowInitialTrajectoryStateJointDiscontinuity_ &&

142  {
143  deltajoint = jointPositions[jointindex] - prevtrajpoint[jointindex];
144

146  {
147  discontinuityDeltaJointIndex = deltajoint;
148  discontinuityJointIndex = jointindex;
149  }
150  }
151  }
152 
153  if (ikAttempts > 0 && discontinuityJointIndex != -1)
154  {
155  k--;
156  ikAttempts--;
157  continue;
158  }
159  else
160  {
161  bool discontinuity = false;
162  if (ikAttempts == 0)
163  {
164  discontinuityIndexes.push_back(k);
165  discontinuity = true;
166  }
167 
168  ikAttempts = 4;
169

171  {
172  // show a message and stop the trajectory generation && jointindex!= 7 || fabs(deltajoint) > 0.1 /*2.5 deg*/ && jointindex== 7
173  std::stringstream ss;
174  ss << "Traj[" << k << "/" << endEffectorTrajectory_.size() << "] "
175  << currentjointnames[discontinuityJointIndex]
176  << " IK discontinuity : " << discontinuityDeltaJointIndex << std::endl
177  << "prev joint value: " << prevtrajpoint[discontinuityJointIndex] << std::endl
178  << "current joint value: " << jointPositions[discontinuityJointIndex] << std::endl;
179 
180  ss << std::endl;
181  for (size_t ji = 0; ji < jointPositions.size(); ji++)
182  {
183  ss << currentjointnames[ji] << ": " << jointPositions[ji] << std::endl;
184  }
185 
186  for (size_t kindex = 0; kindex < trajectory.size(); kindex++)
187  {
188  ss << "[" << kindex << "]: " << trajectory[kindex][discontinuityJointIndex]
189  << std::endl;
190  }
191 
192  if (k == 0)
193  {
194  ss << "This is the first posture of the trajectory. Maybe the robot initial posture is "
195  "not coincident to the initial posture of the generated joint trajectory."
196  << std::endl;
197  }
198 
199  RCLCPP_ERROR_STREAM(getLogger(), ss.str());
200 
201  trajectory.push_back(jointPositions);
202  rclcpp::Duration durationFromStart = rclcpp::Time(pose.header.stamp) - referenceTime;
203  trajectoryTimeStamps.push_back(durationFromStart);
204 
205  continue;
206  }
207  else
208  {
209  trajectory.push_back(jointPositions);
210  rclcpp::Duration durationFromStart = rclcpp::Time(pose.header.stamp) - referenceTime;
211  trajectoryTimeStamps.push_back(durationFromStart);
212 
213  RCLCPP_DEBUG_STREAM(getLogger(), "IK solution: " << res->solution.joint_state);
214  RCLCPP_DEBUG_STREAM(getLogger(), "trajpoint: " << std::endl << ss.str());
215  }
216  }
217  }
218  else
219  {
220  RCLCPP_ERROR(getLogger(), "[CbMoveEndEffectorTrajectory] wrong IK call");
221  }
222  }
223 
224  // interpolate speeds?
225 
226  // interpolate accelerations?
227 
228  // get current robot state
229  // fill plan message
230  // computedMotionPlan.start_state_.joint_state.name = currentjointnames;
231  // computedMotionPlan.start_state_.joint_state.position = trajectory.front();
232  // computedMotionPlan.trajectory_.joint_trajectory.joint_names = currentjointnames;
233 
234  computedJointTrajectory.joint_trajectory.joint_names = currentjointnames;
235  int i = 0;
236  for (auto & p : trajectory)
237  {
238  if (
239  i ==
240  0) // not copy the current state in the trajectory (used to solve discontinuity in other behaviors)
241  {
242  i++;
243  continue;
244  }
245 
246  trajectory_msgs::msg::JointTrajectoryPoint jp;
247  jp.positions = p;
248  jp.time_from_start = trajectoryTimeStamps[i]; //rclcpp::Duration(t);
249  computedJointTrajectory.joint_trajectory.points.push_back(jp);
250  i++;
251  }
252 
253  if (discontinuityIndexes.size())
254  {
255  if (discontinuityIndexes[0] == 0)
256  return ComputeJointTrajectoryErrorCode::INCORRECT_INITIAL_STATE;
257  else
258  return ComputeJointTrajectoryErrorCode::JOINT_TRAJECTORY_DISCONTINUITY;
259  }
260 
261  return ComputeJointTrajectoryErrorCode::SUCCESS;

285 void CbMoveEndEffectorTrajectory::onEntry()
286 {
287  this->requiresClient(movegroupClient_);
288 
289  this->generateTrajectory();
290 
291  if (this->endEffectorTrajectory_.size() == 0)
292  {
293  RCLCPP_WARN_STREAM(
294  getLogger(), "[" << smacc2::demangleSymbol(typeid(*this).name())
295  << "] No points in the trajectory. Skipping behavior.");
296  return;
297  }
298 
299  this->createMarkers();
300  markersInitialized_ = true;
301 
302  moveit_msgs::msg::RobotTrajectory computedTrajectory;
303 
304  RCLCPP_WARN_STREAM(getLogger(), "[" << getName() << "] Computing joint space trajectory.");
305 
306  auto errorcode = computeJointSpaceTrajectory(computedTrajectory);
307 
308  bool trajectoryGenerationSuccess = errorcode == ComputeJointTrajectoryErrorCode::SUCCESS;
309 
310  CpTrajectoryHistory * trajectoryHistory;
311  this->requiresComponent(trajectoryHistory);
312 
313  if (!trajectoryGenerationSuccess)
314  {
315  RCLCPP_INFO_STREAM(
316  getLogger(), "[" << this->getName() << "] Incorrect trajectory. Posting failure event.");
317  if (trajectoryHistory != nullptr)
318  {
319  moveit_msgs::msg::MoveItErrorCodes error;
320  error.val = moveit_msgs::msg::MoveItErrorCodes::NO_IK_SOLUTION;
321  trajectoryHistory->pushTrajectory(this->getName(), computedTrajectory, error);
322  }
323 
324  movegroupClient_->postEventMotionExecutionFailed();
325  this->postFailureEvent();
326 
327  if (errorcode == ComputeJointTrajectoryErrorCode::JOINT_TRAJECTORY_DISCONTINUITY)
328  {
329  this->postJointDiscontinuityEvent(computedTrajectory);
330  }
331  else if (errorcode == ComputeJointTrajectoryErrorCode::INCORRECT_INITIAL_STATE)
332  {
333  this->postIncorrectInitialStateEvent(computedTrajectory);
334  }
335  return;
336  }
337  else
338  {
339  if (trajectoryHistory != nullptr)
340  {
341  moveit_msgs::msg::MoveItErrorCodes error;
342  error.val = moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
343  trajectoryHistory->pushTrajectory(this->getName(), computedTrajectory, error);
344  }
345 
346  this->executeJointSpaceTrajectory(computedTrajectory);
347  }

379 {

392  marker.type = visualization_msgs::msg::Marker::ARROW;

395  marker.scale.y = 0.01;
396  marker.scale.z = 0.01;
397  marker.color.a = 0.8;
398  marker.color.r = 1.0;
399  marker.color.g = 0;
400  marker.color.b = 0;
401 
402  geometry_msgs::msg::Point start, end;
403  start.x = 0;
404  start.y = 0;
405  start.z = 0;
406 
407  tf2::Transform basetransform;
408  tf2::fromMsg(pose.pose, basetransform);
409  // tf2::Transform endarrow = localdirection * basetransform;
410 
411  end.x = localdirection.getOrigin().x();
412  end.y = localdirection.getOrigin().y();
413  end.z = localdirection.getOrigin().z();
414 
415  marker.pose.position = pose.pose.position;
416  marker.pose.orientation = pose.pose.orientation;
417  marker.points.push_back(start);
418  marker.points.push_back(end);
419 
420  beahiorMarkers_.markers.push_back(marker);
421  }
