143 void ForwardLocalPlanner::generateTrajectory(
144  const Eigen::Vector3f & pos, const Eigen::Vector3f & vel, float maxdist, float maxanglediff,
145  float maxtime, float dt, std::vector<Eigen::Vector3f> & outtraj)
146 {
147  // simulate the trajectory and check for collisions, updating costs along the way
148  bool end = false;
149  float time = 0;
150  Eigen::Vector3f currentpos = pos;
151  int i = 0;
152  while (!end)
153  {
154  // add the point to the trajectory so we can draw it later if we want
155  // traj.addPoint(pos[0], pos[1], pos[2]);
156 
157  // if (continued_acceleration_) {
158  // //calculate velocities
159  // loop_vel = computeNewVelocities(sample_target_vel, loop_vel, limits_->getAccLimits(), dt);
160  // //RCLCPP_WARN_NAMED(nh_->get_logger(), "Generator", "Flag: %d, Loop_Vel %f, %f, %f", continued_acceleration_,
161  // loop_vel[0], loop_vel[1], loop_vel[2]);
162  // }
163 
164  auto loop_vel = vel;
165  // update the position of the robot using the velocities passed in
166  auto newpos = computeNewPositions(currentpos, loop_vel, dt);
167 
168  auto dx = newpos[0] - currentpos[0];
169  auto dy = newpos[1] - currentpos[1];
170  float dist, angledist;
171 
172  // RCLCPP_DEBUG(nh_->get_logger(), "traj point %d", i);
173  dist = sqrt(dx * dx + dy * dy);
174  if (dist > maxdist)
175  {
176  end = true;
177  // RCLCPP_DEBUG(nh_->get_logger(), "dist break: %f", dist);
178  }
179  else
180  {
181  // ouble from, double to
182  angledist = fabs(angles::shortest_angular_distance(currentpos[2], newpos[2]));
183  if (angledist > maxanglediff)
184  {
185  end = true;
186  // RCLCPP_DEBUG(nh_->get_logger(), "angle dist break: %f", angledist);
187  }
188  else
189  {
190  outtraj.push_back(newpos);
191 
192  time += dt;
193  if (time > maxtime)
194  {
195  end = true;
196  // RCLCPP_DEBUG(nh_->get_logger(), "time break: %f", time);
197  }
198 
199  // RCLCPP_DEBUG(nh_->get_logger(), "dist: %f, angledist: %f, time: %f", dist, angledist, time);
200  }
201  }
202 
203  currentpos = newpos;
204  i++;
205  } // end for simulation steps
206 }

208 Eigen::Vector3f ForwardLocalPlanner::computeNewPositions(
209  const Eigen::Vector3f & pos, const Eigen::Vector3f & vel, double dt)
210 {
211  Eigen::Vector3f new_pos = Eigen::Vector3f::Zero();
212  new_pos[0] = pos[0] + (vel[0] * cos(pos[2]) + vel[1] * cos(M_PI_2 + pos[2])) * dt;
213  new_pos[1] = pos[1] + (vel[0] * sin(pos[2]) + vel[1] * sin(M_PI_2 + pos[2])) * dt;
214  new_pos[2] = pos[2] + vel[2] * dt;
215  return new_pos;
216 }

259 void clamp(
260  rclcpp::Node::SharedPtr nh_, geometry_msgs::msg::Twist & cmd_vel, double max_linear_x_speed_,
261  double max_angular_z_speed_)
262 {
263  if (max_angular_z_speed_ == 0 || max_linear_x_speed_ == 0) return;
264 
265  if (cmd_vel.angular.z == 0)
266  {
267  cmd_vel.linear.x = max_linear_x_speed_;
268  }
269  else
270  {
271  double kurvature = cmd_vel.linear.x / cmd_vel.angular.z;
272 
273  double linearAuthority = fabs(cmd_vel.linear.x / max_linear_x_speed_);
274  double angularAuthority = fabs(cmd_vel.angular.z / max_angular_z_speed_);
275  if (linearAuthority < angularAuthority)
276  {
277  // lets go to maximum linear speed
278  cmd_vel.linear.x = max_linear_x_speed_;
279  cmd_vel.angular.z = kurvature / max_linear_x_speed_;
280  RCLCPP_WARN_STREAM(
281  nh_->get_logger(), "k=" << kurvature << "lets go to maximum linear capacity: " << cmd_vel);
282  }
283  else
284  {
285  // lets go with maximum angular speed
286  cmd_vel.angular.x = max_angular_z_speed_;
287  cmd_vel.linear.x = kurvature * max_angular_z_speed_;
288  RCLCPP_WARN_STREAM(nh_->get_logger(), "lets go to maximum angular capacity: " << cmd_vel);
289  }
290  }

299 geometry_msgs::msg::TwistStamped ForwardLocalPlanner::computeVelocityCommands(
300  const geometry_msgs::msg::PoseStamped & currentPose, const geometry_msgs::msg::Twist & velocity,
301  nav2_core::GoalChecker * goal_checker)
302 {
303  this->updateParameters();
304 
305  if (this->plan_.size() > 0)
306  {
307  RCLCPP_INFO_STREAM(
308  nh_->get_logger(), "[ForwardLocalPlanner] Current pose frame id: "
309  << plan_.front().header.frame_id
310  << ", path pose frame id: " << currentPose.header.frame_id);
311 
312  if (plan_.front().header.frame_id != currentPose.header.frame_id)
313  {
314  RCLCPP_ERROR_STREAM(nh_->get_logger(), "[ForwardLocalPlanner] Inconsistent frames");
315  }
316  }
317 
318  // xy_goal_tolerance and yaw_goal_tolerance are just used for logging proposes and clamping the carrot
319  // goal distance (parameter safety)
320  if (xy_goal_tolerance_ == -1 || yaw_goal_tolerance_ == -1)
321  {
322  geometry_msgs::msg::Pose posetol;
323  geometry_msgs::msg::Twist twistol;
324  if (goal_checker->getTolerances(posetol, twistol))
325  {
326  xy_goal_tolerance_ = posetol.position.x;
327  yaw_goal_tolerance_ = tf2::getYaw(posetol.orientation);
328  //xy_goal_tolerance_ = posetol.position.x * 0.35; // WORKAROUND DIFFERENCE WITH NAV CONTROLLER GOAL CHECKER
329  //yaw_goal_tolerance_ = tf2::getYaw(posetol.orientation) * 0.35;
330  RCLCPP_INFO_STREAM(
331  nh_->get_logger(), "[ForwardLocalPlanner] xy_goal_tolerance_: " << xy_goal_tolerance_
332  << ", yaw_goal_tolerance_: "
333  << yaw_goal_tolerance_);
334  }
335  else
336  {
337  RCLCPP_INFO_STREAM(
338  nh_->get_logger(), "[ForwardLocalPlanner] could not get tolerances from goal checker");
339  }
340  }
341 
342  geometry_msgs::msg::TwistStamped cmd_vel;
343  goalReached_ = false;
344  RCLCPP_DEBUG(
345  nh_->get_logger(), "[ForwardLocalPlanner] ----- COMPUTE VELOCITY COMMAND LOCAL PLANNER ---");
346 
347  bool ok = false;
348  while (!ok)
349  {
350  // iterate the point from the current position and ahead until reaching a new goal point in the path
351  while (!ok && currentPoseIndex_ < (int)plan_.size())
352  {
353  auto & pose = plan_[currentPoseIndex_];
354  const geometry_msgs::msg::Point & p = pose.pose.position;
355  tf2::Quaternion q;
356  tf2::fromMsg(pose.pose.orientation, q);
357 
358  // take error from the current position to the path point
359  double dx = p.x - currentPose.pose.position.x;
360  double dy = p.y - currentPose.pose.position.y;
361  double dist = sqrt(dx * dx + dy * dy);
362 
363  double pangle = tf2::getYaw(q);
364  double angle = tf2::getYaw(currentPose.pose.orientation);
365  double angular_error = angles::shortest_angular_distance(pangle, angle);
366 
367  if (dist >= carrot_distance_ || fabs(angular_error) > 0.1)
368  {
369  // the target pose is enough different to be defined as a target
370  ok = true;
371  RCLCPP_DEBUG(
372  nh_->get_logger(),
373  "current index: %d, carrot goal percentaje: %lf, dist: %lf, maxdist: %lf, angle_error: "
374  "%lf",
375  currentPoseIndex_, 100.0 * currentPoseIndex_ / plan_.size(), dist, carrot_distance_,
376  angular_error);
377  }
378  else
379  {
380  currentPoseIndex_++;
381  }
382  }
383 
384  RCLCPP_DEBUG_STREAM(
385  nh_->get_logger(), "[ForwardLocalPlanner] selected carrot pose index "
386  << currentPoseIndex_ << "/" << plan_.size());
387 
388  if (currentPoseIndex_ >= (int)plan_.size())
389  {
390  // even the latest point is quite similar, then take the last since it is the final goal
391  cmd_vel.twist.linear.x = 0;
392  cmd_vel.twist.angular.z = 0;
393  // RCLCPP_INFO(nh_->get_logger(), "End Local planner");
394  ok = true;
395  currentPoseIndex_ = (int)plan_.size() - 1;
396  // return true;
397  }
398  }
399 
400  // RCLCPP_INFO(nh_->get_logger(), "pose control algorithm");
401 
402  const geometry_msgs::msg::PoseStamped & finalgoalpose = plan_.back();
403  const geometry_msgs::msg::PoseStamped & carrot_goalpose = plan_[currentPoseIndex_];
404  const geometry_msgs::msg::Point & goalposition = carrot_goalpose.pose.position;
405 
406  tf2::Quaternion carrotGoalQ;
407  tf2::fromMsg(carrot_goalpose.pose.orientation, carrotGoalQ);
408  // RCLCPP_INFO_STREAM(nh_->get_logger(), "Plan goal quaternion at "<< carrot_goalpose.pose.orientation);
409 
410  // goal orientation (global frame)
411  double betta = tf2::getYaw(carrot_goalpose.pose.orientation) + betta_offset_;
412  double dx = goalposition.x - currentPose.pose.position.x;
413  double dy = goalposition.y - currentPose.pose.position.y;
414 
415  // distance error to the targetpoint
416  double rho_error = sqrt(dx * dx + dy * dy);
417 
418  tf2::Quaternion currentOrientation;
419  tf2::fromMsg(currentPose.pose.orientation, currentOrientation);
420 
421  // current angle
422  double theta = tf2::getYaw(currentOrientation);
423  double alpha = atan2(dy, dx);
424  alpha = alpha + alpha_offset_;
425 
426  double alpha_error = angles::shortest_angular_distance(alpha, theta);
427  double betta_error = angles::shortest_angular_distance(betta, theta);
428 
429  double vetta = 0; // = k_rho_ * rho_error;
430  double gamma = 0; //= k_alpha_ * alpha_error + k_betta_ * betta_error;
431 
432  if (
433  rho_error >
434  xy_goal_tolerance_) // reguular control rule, be careful, rho error is with the carrot not with the
435  // final goal (this is something to improve like the backwards planner)
436  {
437  vetta = k_rho_ * rho_error;
438  gamma = k_alpha_ * alpha_error;
439  }
440  else if (fabs(betta_error) >= yaw_goal_tolerance_) // pureSpining
441  {
442  vetta = 0;
443  gamma = k_betta_ * betta_error;
444  }
445  else // goal reached
446  {
447  RCLCPP_DEBUG(nh_->get_logger(), "GOAL REACHED");
448  vetta = 0;
449  gamma = 0;
450  goalReached_ = true;
451  }
452 
453  // linear speed clamp
454  if (vetta > max_linear_x_speed_)
455  {
456  vetta = max_linear_x_speed_;
457  }
458  else if (vetta < -max_linear_x_speed_)
459  {
460  vetta = -max_linear_x_speed_;
461  }
462 
463  // angular speed clamp
464  if (gamma > max_angular_z_speed_)
465  {
466  gamma = max_angular_z_speed_;
467  }
468  else if (gamma < -max_angular_z_speed_)
469  {
470  gamma = -max_angular_z_speed_;
471  }
472 
473  cmd_vel.twist.linear.x = vetta;
474  cmd_vel.twist.angular.z = gamma;
475 
476  // clamp(cmd_vel, max_linear_x_speed_, max_angular_z_speed_);
477 
478  // RCLCPP_INFO_STREAM(nh_->get_logger(), "Local planner: "<< cmd_vel);
479 
480  publishGoalMarker(goalposition.x, goalposition.y, betta);
481 
482  RCLCPP_DEBUG_STREAM(
483  nh_->get_logger(), "Forward local planner,"
484  << std::endl
485  << " theta: " << theta << std::endl
486  << " betta: " << betta << std::endl
487  << " err_x: " << dx << std::endl
488  << " err_y:" << dy << std::endl
489  << " rho_error:" << rho_error << std::endl
490  << " alpha_error:" << alpha_error << std::endl
491  << " betta_error:" << betta_error << std::endl
492  << " vetta:" << vetta << std::endl
493  << " gamma:" << gamma << std::endl
494  << " xy_goal_tolerance:" << xy_goal_tolerance_ << std::endl
495  << " yaw_goal_tolerance:" << yaw_goal_tolerance_ << std::endl);
496 
497  // if(cmd_vel.linear.x==0 && cmd_vel.angular.z == 0 )
498  //{
499  //}
500 
501  // integrate trajectory and check collision
502 
503  assert(currentPose.header.frame_id == "odom" || currentPose.header.frame_id == "map");
504  auto global_pose = currentPose;
505  //->getRobotPose(global_pose);
506 
507  auto * costmap2d = costmapRos_->getCostmap();
508  auto yaw = tf2::getYaw(global_pose.pose.orientation);
509 
510  auto & pos = global_pose.pose.position;
511 
512  Eigen::Vector3f currentpose(pos.x, pos.y, yaw);
513  Eigen::Vector3f currentvel(
514  cmd_vel.twist.linear.x, cmd_vel.twist.linear.y, cmd_vel.twist.angular.z);
515 
516  std::vector<Eigen::Vector3f> trajectory;
517  this->generateTrajectory(
518  currentpose, currentvel, 0.8 /*meters*/, M_PI / 8 /*rads*/, 3.0 /*seconds*/, 0.05 /*seconds*/,
519  trajectory);
520 
521  // check plan rejection
522  bool aceptedplan = true;
523 
524  unsigned int mx, my;
525 
526  int i = 0;
527  // RCLCPP_INFO_STREAM(nh_->get_logger(), "lplanner goal: " << finalgoalpose.pose.position);
528  for (auto & p : trajectory)
529  {
530  float dx = p[0] - finalgoalpose.pose.position.x;
531  float dy = p[1] - finalgoalpose.pose.position.y;
532 
533  float dst = sqrt(dx * dx + dy * dy);
534  if (dst < xy_goal_tolerance_)
535  {
536  // RCLCPP_INFO(nh_->get_logger(), "trajectory checking skipped, goal reached");
537  break;
538  }
539 
540  costmap2d->worldToMap(p[0], p[1], mx, my);
541 
542  // RCLCPP_INFO(nh_->get_logger(), "checking cost pt %d [%lf, %lf] cell[%d,%d] = %d", i, p[0], p[1], mx, my, cost);
543  // RCLCPP_INFO_STREAM(nh_->get_logger(), "cost: " << cost);
544 
545  // static const unsigned char NO_INFORMATION = 255;
546  // static const unsigned char LETHAL_OBSTACLE = 254;
547  // static const unsigned char INSCRIBED_INFLATED_OBSTACLE = 253;
548  // static const unsigned char FREE_SPACE = 0;
549 
550  if (costmap2d->getCost(mx, my) >= nav2_costmap_2d::INSCRIBED_INFLATED_OBSTACLE)
551  {
552  aceptedplan = false;
553  // RCLCPP_WARN(nh_->get_logger(), "ABORTED LOCAL PLAN BECAUSE OBSTACLE DETEDTED");
554  break;
555  }
556  i++;
557  }
558 
559  bool success = false;
560  if (aceptedplan)
561  {
562  waiting_ = false;
563  success = true;
564  RCLCPP_DEBUG(nh_->get_logger(), "simulated trajectory is accepted.");
565  }
566  else
567  {
568  RCLCPP_DEBUG(nh_->get_logger(), "simulated trajectory is not accepted. Stop command.");
569 
570  // stop and wait
571  cmd_vel.twist.linear.x = 0;
572  cmd_vel.twist.angular.z = 0;
573 
574  if (!waiting_)
575  {
576  RCLCPP_DEBUG(nh_->get_logger(), "Start waiting obstacle disappear");
577  waiting_ = true;
578  waitingStamp_ = nh_->now();
579  }
580  else
581  {
582  auto waitingduration = nh_->now() - waitingStamp_;
583  RCLCPP_DEBUG(
584  nh_->get_logger(), "waiting obstacle disappear, elapsed: %lf seconds",
585  waitingduration.seconds());
586 
587  if (waitingduration > this->waitingTimeout_)
588  {
589  RCLCPP_WARN(
590  nh_->get_logger(), "TIMEOUT waiting obstacle disappear, elapsed: %lf seconds",
591  waitingduration.seconds());
592  success = false;
593  }
594  }
595  }
596 
597  if (!success)
598  {
