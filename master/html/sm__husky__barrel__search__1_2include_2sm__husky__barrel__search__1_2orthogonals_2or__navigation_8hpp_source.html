77 void loadWaypointsFromYaml(WaypointNavigator * waypointsNavigator)
78 {
79 RCLCPP_INFO(getLogger(), "OrNavigation loading parameters file");
80 // if it is the first time and the waypoints navigator is not configured
81 std::string planfilepath;
82 getNode()->declare_parameter("waypoints_plan", planfilepath);
83 if (getNode()->get_parameter("waypoints_plan", planfilepath))
84 {
85 std::string package_share_directory =
86 ament_index_cpp::get_package_share_directory("sm_husky_barrel_search_1");
87 boost::replace_all(planfilepath, "$(pkg_share)", package_share_directory);
88 waypointsNavigator->loadWayPointsFromFile(planfilepath);
89 RCLCPP_INFO(getLogger(), "waypoints plan: %s", planfilepath.c_str());
90 }
91 else
92 {
93 RCLCPP_ERROR(getLogger(), "waypoints plan file not found: NONE");
