41 void CbNavigateForward::onEntry()
42 {
43  // straight motion distance
44  double dist = 3.5;
45 
46  if (!forwardDistance)
47  {
48  // second choice parameter server
49  if (!this->getCurrentState()->getParam("straight_motion_distance", dist))
50  {
51  // last choice default value;
52  }
53  }
54  else
55  {
56  dist = *forwardDistance;
57  }
58 
59  RCLCPP_INFO_STREAM(getLogger(), "[CbNavigateForward] Straight motion distance: " << dist);
60 
61  // get current pose
62  auto p = moveBaseClient_->getComponent<cl_move_base_z::Pose>();
63  auto referenceFrame = p->getReferenceFrame();
64  auto currentPoseMsg = p->toPoseMsg();
65  tf2::Transform currentPose;
66  tf2::fromMsg(currentPoseMsg, currentPose);
67 
68  RCLCPP_INFO_STREAM(getLogger(), "[CbNavigateForward] current pose: " << currentPoseMsg);
69 
70  // force global orientation if it is requested
71  if (this->forceInitialOrientation)
72  {
73  currentPoseMsg.orientation = *forceInitialOrientation;
74  RCLCPP_WARN_STREAM(
75  getLogger(), "[CbNavigateForward] Forcing initial straight motion orientation: "
76  << currentPoseMsg.orientation);
77  }
78 
79  // compute forward goal pose
80  tf2::Transform forwardDeltaTransform;
81  forwardDeltaTransform.setIdentity();
82  forwardDeltaTransform.setOrigin(tf2::Vector3(dist, 0, 0));
83 
84  tf2::Transform targetPose = currentPose * forwardDeltaTransform;
85 
86  ClMoveBaseZ::Goal goal;
87  goal.pose.header.frame_id = referenceFrame;
88  goal.pose.header.stamp = getNode()->now();
89  tf2::toMsg(targetPose, goal.pose.pose);
90 
91  RCLCPP_INFO_STREAM(getLogger(), "TARGET POSE FORWARD: " << goal.pose.pose);
92 
93  geometry_msgs::msg::PoseStamped currentStampedPoseMsg;
94  currentStampedPoseMsg.header.frame_id = referenceFrame;
95  currentStampedPoseMsg.header.stamp = getNode()->now();
96  tf2::toMsg(currentPose, currentStampedPoseMsg.pose);
97 
98  odomTracker_ = moveBaseClient_->getComponent<OdomTracker>();
99  odomTracker_->pushPath();
100 
101  odomTracker_->setStartPoint(currentStampedPoseMsg);
102  odomTracker_->setWorkingMode(WorkingMode::RECORD_PATH);
103 
104  auto plannerSwitcher = moveBaseClient_->getComponent<PlannerSwitcher>();
105  plannerSwitcher->setForwardPlanner();
106 
107  auto goalCheckerSwitcher = moveBaseClient_->getComponent<GoalCheckerSwitcher>();
108  goalCheckerSwitcher->setGoalCheckerId("forward_goal_checker");
109 
110  moveBaseClient_->sendGoal(goal);
111 }
