159 inline double p2pDistance(
160  const geometry_msgs::msg::Point & p1, const geometry_msgs::msg::Point & p2)
161 {
162  double dx = (p1.x - p2.x);
163  double dy = (p1.y - p2.y);
164  double dz = (p2.z - p2.z);
165  double dist = sqrt(dx * dx + dy * dy + dz * dz);
166  return dist;
167 }
