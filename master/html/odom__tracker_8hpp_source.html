157 inline double p2pDistance(
158  const geometry_msgs::msg::Point & p1, const geometry_msgs::msg::Point & p2)
159 {
160  double dx = (p1.x - p2.x);
161  double dy = (p1.y - p2.y);
162  double dz = (p2.z - p2.z);
163  double dist = sqrt(dx * dx + dy * dy + dz * dz);
164  return dist;
165 }
