68 void WaypointNavigator::sendNextGoal()
69 {
70  if (currentWaypoint_ >= 0 && currentWaypoint_ < (int)waypoints_.size())
71  {
72  auto & next = waypoints_[currentWaypoint_];
73 
74  ClMoveBaseZ::Goal goal;
75  auto p = client_->getComponent<cl_move_base_z::Pose>();
76  auto pose = p->toPoseMsg();
77 
78  // configuring goal
79  goal.pose.header.frame_id = p->getReferenceFrame();
80  goal.pose.header.stamp = getNode()->now();
81  goal.pose.pose = next;
82 
83  RCLCPP_WARN(getLogger(), "[WaypointsNavigator] Configuring default planners");
84  auto plannerSwitcher = client_->getComponent<PlannerSwitcher>();
85  plannerSwitcher->setDefaultPlanners();
86 
87  RCLCPP_WARN(getLogger(), "[WaypointsNavigator] Configuring default goal planner");
88  auto goalCheckerSwitcher = client_->getComponent<GoalCheckerSwitcher>();
89  goalCheckerSwitcher->setGoalCheckerId("goal_checker");
90 
91  // publish stuff
92  //rclcpp::sleep_for(5s);
93 
94  RCLCPP_INFO(getLogger(), "[WaypointsNavigator] Getting odom tracker");
95  auto odomTracker = client_->getComponent<cl_move_base_z::odom_tracker::OdomTracker>();
96  if (odomTracker != nullptr)
97  {
98  RCLCPP_INFO(getLogger(), "[WaypointsNavigator] Storing path in odom tracker");
99  odomTracker->pushPath();
100  odomTracker->setStartPoint(pose);
101  odomTracker->setWorkingMode(cl_move_base_z::odom_tracker::WorkingMode::RECORD_PATH);
102  }
103 
104  // SEND GOAL
105  if (!succeddedConnection_.connected())
106  {
107  this->succeddedConnection_ = client_->onSucceeded(&WaypointNavigator::onGoalReached, this);
108  this->cancelledConnection_ = client_->onAborted(&WaypointNavigator::onGoalCancelled, this);
109  this->abortedConnection_ = client_->onCancelled(&WaypointNavigator::onGoalAborted, this);
110  }
111 
112  client_->sendGoal(goal);
113  }
114  else
115  {
116  RCLCPP_WARN(
117  getLogger(),
118  "[WaypointsNavigator] All waypoints were consumed. There is no more waypoints available.");
119  }
120 }

135 void WaypointNavigator::setWaypoints(const std::vector<Pose2D> & waypoints)
136 {
137  waypoints_.clear();
138  waypointsNames_.clear();
139  int i = 0;
140  for (auto & p : waypoints)
141  {
142  geometry_msgs::msg::Pose pose;
143  pose.position.x = p.x_;
144  pose.position.y = p.y_;
145  pose.position.z = 0.0;
146  tf2::Quaternion q;
147  q.setRPY(0, 0, p.yaw_);
148  pose.orientation = tf2::toMsg(q);
149 
150  waypoints_.push_back(pose);
151  waypointsNames_.push_back(std::to_string(i++));
152  }
153 }

171 void WaypointNavigator::loadWayPointsFromFile(std::string filepath)
172 {
173  RCLCPP_INFO_STREAM(getLogger(), "[WaypointNavigator] Loading file:" << filepath);
174  this->waypoints_.clear();
175  std::ifstream ifs(filepath.c_str(), std::ifstream::in);
176  if (ifs.good() == false)
177  {
178  throw std::string("Waypoints file not found");
179  }
180 
181  try
182  {
183 #ifdef HAVE_NEW_YAMLCPP
184  YAML::Node node = YAML::Load(ifs);
185 #else
186  YAML::Parser parser(ifs);
187  parser.GetNextDocument(node);
188 #endif
189 
190 #ifdef HAVE_NEW_YAMLCPP
191  const YAML::Node & wp_node_tmp = node["waypoints"];
192  const YAML::Node * wp_node = wp_node_tmp ? &wp_node_tmp : NULL;
193 #else
194  const YAML::Node * wp_node = node.FindValue("waypoints");
195 #endif
196 
197  if (wp_node != NULL)
198  {
199  for (unsigned int i = 0; i < wp_node->size(); ++i)
200  {
201  // Parse waypoint entries on YAML
202  geometry_msgs::msg::Pose wp;
203 
204  try
205  {
206  // (*wp_node)[i]["name"] >> wp.name;
207  // (*wp_node)[i]["frame_id"] >> wp.header.frame_id;
208  wp.position.x = (*wp_node)[i]["position"]["x"].as<double>();
209  wp.position.y = (*wp_node)[i]["position"]["y"].as<double>();
210  wp.position.z = (*wp_node)[i]["position"]["z"].as<double>();
211  wp.orientation.x = (*wp_node)[i]["orientation"]["x"].as<double>();
212  wp.orientation.y = (*wp_node)[i]["orientation"]["y"].as<double>();
213  wp.orientation.z = (*wp_node)[i]["orientation"]["z"].as<double>();
214  wp.orientation.w = (*wp_node)[i]["orientation"]["w"].as<double>();
215 
216  this->waypoints_.push_back(wp);
217  }
218  catch (...)
219  {
220  RCLCPP_ERROR(getLogger(), "parsing waypoint file, syntax error in point %d", i);
221  }
222  }
223  RCLCPP_INFO_STREAM(getLogger(), "Parsed " << this->waypoints_.size() << " waypoints.");
224  }
225  else
226  {
227  RCLCPP_WARN_STREAM(getLogger(), "Couldn't find any waypoints in the provided yaml file.");
228  }
229  }
230  catch (const YAML::ParserException & ex)
231  {
232  RCLCPP_ERROR_STREAM(
233  getLogger(), "Error loading the Waypoints YAML file. Incorrect syntax: " << ex.what());
234  }
