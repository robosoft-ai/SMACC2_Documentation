33 // Asynchronous behavior that waits to a topic message to send EvCbSuccess event
34 // a guard function can be set to use conditions on the contents
35 template <typename TMessage>
36 class CbWaitTopicMessage : public smacc2::SmaccAsyncClientBehavior
37 {
38 public:
39  CbWaitTopicMessage(
40  const char * topicname, std::function<bool(const TMessage &)> checkfunction = nullptr)
41  {
42  topicname_ = topicname;
43  checkFn_ = checkfunction;
44  }
45 
46  void onEntry() override
47  {
48  rclcpp::SensorDataQoS qos;
49  rclcpp::SubscriptionOptions sub_option;
50 
51  sub_ = getNode()->create_subscription<TMessage>(
52  topicname_, qos,
53  std::bind(&CbWaitTopicMessage<TMessage>::onMessageReceived, this, std::placeholders::_1),
54  sub_option);
55  }
56 
57  void onMessageReceived(const typename TMessage::SharedPtr msg)
58  {
59  if (checkFn_ == nullptr || checkFn_(*msg))
60  {
61  RCLCPP_INFO(getLogger(), "[CbWaitTopicMessage] message received.");
62  success = true;
63  this->postSuccessEvent();
64  }
65  }
66 
67 protected:

69  typename rclcpp::Subscription<TMessage>::SharedPtr sub_;
