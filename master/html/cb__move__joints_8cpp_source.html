49 std::string currentJointStatesToString(
50  moveit::planning_interface::MoveGroupInterface & moveGroupInterface,
51  std::map<std::string, double> & targetJoints)
52 {
53  auto state = moveGroupInterface.getCurrentState();
54  auto vnames = state->getVariableNames();
55 
56  std::stringstream ss;
57 
58  for (auto & tgj : targetJoints)
59  {
60  auto it = std::find(vnames.begin(), vnames.end(), tgj.first);
61  auto index = std::distance(vnames.begin(), it);

63  ss << tgj.first << ":" << state->getVariablePosition(index) << std::endl;

69 void CbMoveJoints::moveJoints(moveit::planning_interface::MoveGroupInterface & moveGroupInterface)
70 {
71  if (scalingFactor_) moveGroupInterface.setMaxVelocityScalingFactor(*scalingFactor_);
72 
73  bool success;
74  moveit::planning_interface::MoveGroupInterface::Plan computedMotionPlan;
75 
76  if (jointValueTarget_.size() == 0)
77  {
78  RCLCPP_WARN(
79  getLogger(), "[CbMoveJoints] No joint was value specified. Skipping planning call.");
80  success = false;
81  }
82  else
83  {
84  moveGroupInterface.setJointValueTarget(jointValueTarget_);
85  //moveGroupInterface.setGoalJointTolerance(0.01);
86  success =
87  (moveGroupInterface.plan(computedMotionPlan) ==
88  moveit::planning_interface::MoveItErrorCode::SUCCESS);
89  RCLCPP_INFO(getLogger(), "Success Visualizing plan 1 (pose goal) %s", success ? "" : "FAILED");
90  }
91 
92  if (success)
93  {
94  auto executionResult = moveGroupInterface.execute(computedMotionPlan);
95 
96  auto statestr = currentJointStatesToString(moveGroupInterface, jointValueTarget_);
97 
98  if (executionResult == moveit_msgs::msg::MoveItErrorCodes::SUCCESS)
99  {
100  RCLCPP_INFO_STREAM(
101  getLogger(), "[" << this->getName()
102  << "] motion execution succeeded. Throwing success event. " << std::endl
103  << statestr);
104  movegroupClient_->postEventMotionExecutionSucceded();
105  this->postSuccessEvent();
106  }
107  else
108  {
109  RCLCPP_WARN_STREAM(
110  getLogger(), "[" << this->getName() << "] motion execution failed. Throwing fail event."
111  << std::endl
112  << statestr);
113  movegroupClient_->postEventMotionExecutionFailed();
114  this->postFailureEvent();
115  }
116  }
117  else
118  {
119  auto statestr = currentJointStatesToString(moveGroupInterface, jointValueTarget_);
120  RCLCPP_WARN_STREAM(
121  getLogger(), "[" << this->getName() << "] motion execution failed. Throwing fail event."
122  << std::endl
123  << statestr);
124  movegroupClient_->postEventMotionExecutionFailed();
125  this->postFailureEvent();
126  }
127 }
