227  template <
228  typename TStateReactor, typename TOutputEvent, typename TInputEventList, typename... TArgs>
229  static std::shared_ptr<smacc2::introspection::StateReactorHandler> static_createStateReactor(
230  TArgs... args)
231  {
232  auto srh = std::make_shared<smacc2::introspection::StateReactorHandler>(globalNh_);
233  auto srinfo = std::make_shared<SmaccStateReactorInfo>();
234 
235  srinfo->stateReactorType = TypeInfo::getTypeInfoFromType<TStateReactor>();
236  srinfo->outputEventType = TypeInfo::getTypeInfoFromType<TOutputEvent>();
237 
238  if (srinfo->outputEventType->templateParameters.size() == 2)
239  {
240  srinfo->objectTagType = srinfo->outputEventType->templateParameters[1];
241  }
242  else if (srinfo->outputEventType->templateParameters.size() == 1)
243  {
244  srinfo->objectTagType = TypeInfo::getTypeInfoFromType<state_reactors::EmptyObjectTag>();
245  }
246  else
247  {
248  assert(
249  false &&
250  "state reactor output events should have one or two parameters (SourceType, ObjectTag)");
251  }
252 
253  // iterate statically on all event sources
254  using boost::mpl::_1;
255  using wrappedList = typename boost::mpl::transform<TInputEventList, _1>::type;
256  AddTEventTypeStateReactorInfo<TInputEventList> op(srinfo.get());
257  boost::mpl::for_each<wrappedList>(op);
258 
259  srinfo->srh = srh;
260  srh->srInfo_ = srinfo;
261 
262  const std::type_info * tindex = &(typeid(MostDerived)); // get identifier of the current state
263 
264  if (!SmaccStateInfo::stateReactorsInfo.count(tindex))
265  SmaccStateInfo::stateReactorsInfo[tindex] =
266  std::vector<std::shared_ptr<SmaccStateReactorInfo>>();
267 
268  srinfo->factoryFunction = [&, srh, args...](ISmaccState * state) {
269  auto sr =
270  state->createStateReactor<TStateReactor, TOutputEvent, TInputEventList, TArgs...>(args...);
271  srh->configureStateReactor(sr);
272  sr->initialize(state);
273  return sr;
274  };
275 
276  SmaccStateInfo::stateReactorsInfo[tindex].push_back(srinfo);
277 
278  return srh;
279  }

424  void entryStateInternal()
425  {
426 #define STATE_NAME (demangleSymbol(typeid(MostDerived).name()).c_str())
427  // finally we go to the derived state onEntry Function
428 
429  RCLCPP_INFO(
430  getNode()->get_logger(), "[%s] State object created. Initializating...", STATE_NAME);
431  this->getStateMachine().notifyOnStateEntryStart(static_cast<MostDerived *>(this));
432 
433  // TODO: make this static to build the parameter tree at startup
434  // this->getNode() = rclcpp::Node::make_shared(std::string(contextNh->get_namespace()) + std::string("/") +
435  // smacc2::utils::cleanShortTypeName(typeid(MostDerived)).c_str());
436  RCLCPP_INFO_STREAM(
437  this->getNode()->get_logger(),
438  "[" << smacc2::utils::cleanShortTypeName(typeid(MostDerived)).c_str()
439  << "] creating ros subnode");
440 
441  RCLCPP_DEBUG(
442  getNode()->get_logger(), "[%s] nodehandle namespace: %s", STATE_NAME,
443  getNode()->get_namespace());
444 
445  bool created;
446  if (!this->getParam("visited", created))
447  {
448  this->param("visited");
449  this->setParam("visited", true);
450  }
451 
452  // before dynamic runtimeConfigure, we execute the staticConfigure behavior configurations
453  {
454  RCLCPP_INFO(getNode()->get_logger(), "[%s] -- STATIC STATE DESCRIPTION --", STATE_NAME);
455 
456  for (const auto & stateReactorsVector : SmaccStateInfo::staticBehaviorInfo)
457  {
458  RCLCPP_DEBUG(
459  getNode()->get_logger(), "[%s] state reactor info: %s", STATE_NAME,
460  demangleSymbol(stateReactorsVector.first->name()).c_str());
461  for (auto & srinfo : stateReactorsVector.second)
462  {
463  RCLCPP_DEBUG(
464  getNode()->get_logger(), "[%s] state reactor: %s", STATE_NAME,
465  demangleSymbol(srinfo.behaviorType->name()).c_str());
466  }
467  }
468 
469  const std::type_info * tindex = &(typeid(MostDerived));
470  auto & staticDefinedBehaviors = SmaccStateInfo::staticBehaviorInfo[tindex];
471  auto & staticDefinedStateReactors = SmaccStateInfo::stateReactorsInfo[tindex];
472  auto & staticDefinedEventGenerators = SmaccStateInfo::eventGeneratorsInfo[tindex];
473 
474  RCLCPP_DEBUG_STREAM(
475  getNode()->get_logger(), "finding static client behaviors. State Database: "
476  << SmaccStateInfo::staticBehaviorInfo.size()
477  << ". Current state " << cleanShortTypeName(*tindex) << " cbs: "
478  << SmaccStateInfo::staticBehaviorInfo[tindex].size());
479  for (auto & bhinfo : staticDefinedBehaviors)
480  {
481  RCLCPP_INFO(
482  getNode()->get_logger(), "[%s] Creating static client behavior: %s", STATE_NAME,
483  demangleSymbol(bhinfo.behaviorType->name()).c_str());
484  bhinfo.factoryFunction(this);
485  }
486 
487  for (auto & sr : staticDefinedStateReactors)
488  {
489  RCLCPP_INFO(
490  getNode()->get_logger(), "[%s] Creating static state reactor: %s", STATE_NAME,
491  sr->stateReactorType->getFullName().c_str());
492  sr->factoryFunction(this);
493  }
494 
495  for (auto & eg : staticDefinedEventGenerators)
496  {
497  RCLCPP_INFO(
498  getNode()->get_logger(), "[%s] Creating static event generator: %s", STATE_NAME,
499  eg->eventGeneratorType->getFullName().c_str());
500  eg->factoryFunction(this);
501  }
502 
503  RCLCPP_INFO(getNode()->get_logger(), "[%s] ---- END STATIC DESCRIPTION", STATE_NAME);
504  }
505 
506  RCLCPP_INFO(getNode()->get_logger(), "[%s] State runtime configuration", STATE_NAME);
507 
508  auto * derivedthis = static_cast<MostDerived *>(this);
509 
510  // second the orthogonals are internally configured
511  this->getStateMachine().notifyOnRuntimeConfigured(derivedthis);
512 
513  TRACEPOINT(smacc2_state_onRuntimeConfigure_start, STATE_NAME);
514  // first we runtime configure the state, where we create client behaviors
515  static_cast<MostDerived *>(this)->runtimeConfigure();
516  TRACEPOINT(smacc2_state_onRuntimeConfigure_end, STATE_NAME);
517 
518  this->getStateMachine().notifyOnRuntimeConfigurationFinished(derivedthis);
519 
520  RCLCPP_INFO(getNode()->get_logger(), "[%s] State OnEntry", STATE_NAME);
521 
522  static_cast<MostDerived *>(this)->onEntry();
523 
524  // here orthogonals and client behaviors are entered OnEntry
525  this->getStateMachine().notifyOnStateEntryEnd(derivedthis);
526  }
