86 geometry_msgs::msg::TwistStamped PureSpinningLocalPlanner::computeVelocityCommands(
87  const geometry_msgs::msg::PoseStamped & pose, const geometry_msgs::msg::Twist & velocity,
88  nav2_core::GoalChecker * goal_checker)
89 {
90  this->updateParameters();
91  if (yaw_goal_tolerance_ == -1)
92  {
93  geometry_msgs::msg::Pose posetol;
94  geometry_msgs::msg::Twist twistol;
95  if (goal_checker->getTolerances(posetol, twistol))
96  {
97  yaw_goal_tolerance_ = tf2::getYaw(posetol.orientation);
98  //yaw_goal_tolerance_ = tf2::getYaw(posetol.orientation) * 0.35; // WORKAROUND GOAL CHECKER DIFFERENCE NAV CONTROLLER
99  RCLCPP_INFO_STREAM(
100  nh_->get_logger(),
101  "[PureSpinningLocalPlanner] yaw_goal_tolerance_: " << yaw_goal_tolerance_);
102  }
103  else
104  {
105  RCLCPP_INFO_STREAM(
106  nh_->get_logger(), "[PureSpinningLocalPlanner] could not get tolerances from goal checker");
107  }
108  }
109 
110  geometry_msgs::msg::TwistStamped cmd_vel;
111 
112  goalReached_ = false;
113  // RCLCPP_DEBUG(nh_->get_logger(), "LOCAL PLANNER LOOP");
114 
115  geometry_msgs::msg::PoseStamped currentPose = pose;
116 
117  tf2::Quaternion q;
118  tf2::fromMsg(currentPose.pose.orientation, q);
119  auto currentYaw = tf2::getYaw(currentPose.pose.orientation);
120  double angular_error = 0;
121  double targetYaw;
122 
123  do
124  {
125  auto & goal = plan_[currentPoseIndex_];
126  targetYaw = tf2::getYaw(goal.pose.orientation);
127 
128  angular_error = angles::shortest_angular_distance(currentYaw, targetYaw);
129 
130  // if it is in the following way, sometimes the direction is incorrect
131  //angular_error = targetYaw - currentYaw;
132 
133  // all the points must be reached using the control rule, but the last one
134  // have an special condition
135  if ((currentPoseIndex_ < (int)plan_.size() - 1 &&
136  fabs(angular_error) < this->intermediate_goal_yaw_tolerance_))
137  {
138  currentPoseIndex_++;
139  }
140  else
141  {
142  break;
143  }
144  } while (currentPoseIndex_ < (int)plan_.size());
145 
146  auto omega = angular_error * k_betta_;
147  cmd_vel.twist.angular.z =
148  std::min(std::max(omega, -fabs(max_angular_z_speed_)), fabs(max_angular_z_speed_));
149 
150  RCLCPP_INFO_STREAM(
151  nh_->get_logger(),
152  "[PureSpinningLocalPlanner] pose index: " << currentPoseIndex_ << "/" << plan_.size());
153  RCLCPP_INFO_STREAM(nh_->get_logger(), "[PureSpinningLocalPlanner] k_betta_: " << k_betta_);
154  RCLCPP_INFO_STREAM(
155  nh_->get_logger(), "[PureSpinningLocalPlanner] angular error: " << angular_error << "("
156  << yaw_goal_tolerance_ << ")");
157  RCLCPP_INFO_STREAM(
158  nh_->get_logger(),
159  "[PureSpinningLocalPlanner] command angular speed: " << cmd_vel.twist.angular.z);
160  RCLCPP_INFO_STREAM(
161  nh_->get_logger(),
162  "[PureSpinningLocalPlanner] completion " << currentPoseIndex_ << "/" << plan_.size());
163 
164  if (currentPoseIndex_ >= (int)plan_.size() - 1 && fabs(angular_error) < yaw_goal_tolerance_)
165  {
166  RCLCPP_INFO_STREAM(
167  nh_->get_logger(), "[PureSpinningLocalPlanner] GOAL REACHED. Sending stop command.");
168  goalReached_ = true;
169  cmd_vel.twist.linear.x = 0;
170  cmd_vel.twist.angular.z = 0;
171  }
172 
173  return cmd_vel;
174 }

186 void PureSpinningLocalPlanner::setPlan(const nav_msgs::msg::Path & path)
187 {
188  RCLCPP_INFO_STREAM(nh_->get_logger(), "activating controller PureSpinningLocalPlanner");
189 
190  nav_msgs::msg::Path transformedPlan;
191 
192  rclcpp::Duration ttol = rclcpp::Duration::from_seconds(transform_tolerance_);
193  // transform global plan
194  for (auto & p : path.poses)
195  {
196  geometry_msgs::msg::PoseStamped transformedPose;
197  nav_2d_utils::transformPose(tf_, costmapRos_->getGlobalFrameID(), p, transformedPose, ttol);
198  transformedPose.header.frame_id = costmapRos_->getGlobalFrameID();
199  transformedPlan.poses.push_back(transformedPose);
200  }
201 
202  plan_ = transformedPlan.poses;
203 
204  goalReached_ = false;
205  currentPoseIndex_ = 0;
206 }
