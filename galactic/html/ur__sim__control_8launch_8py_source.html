51def launch_setup(context, *args, **kwargs):
52
53 # Initialize Arguments
54 ur_type = LaunchConfiguration("ur_type")
55 safety_limits = LaunchConfiguration("safety_limits")
56 safety_pos_margin = LaunchConfiguration("safety_pos_margin")
57 safety_k_position = LaunchConfiguration("safety_k_position")
58 # General arguments
59 # runtime_config_package = LaunchConfiguration("runtime_config_package")
60 # controllers_file = LaunchConfiguration("controllers_file")
61 description_package = LaunchConfiguration("description_package")
62 description_file = LaunchConfiguration("description_file")
63 prefix = LaunchConfiguration("prefix")
64 prefixvalue = prefix.perform(context)
65
66 start_joint_controller = LaunchConfiguration("start_joint_controller")
67 # initial_joint_controller = LaunchConfiguration("initial_joint_controller")
68
69 initial_joint_controller = "joint_trajectory_controller" + "_" + prefix.perform(context)
70
71 x = LaunchConfiguration("x")
72 y = LaunchConfiguration("y")
73 z = LaunchConfiguration("z")
74
75 initial_joint_controllers = PathJoinSubstitution(
76 [
77 FindPackageShare("sm_multi_ur5_sim"),
78 "config",
79 "ros_control",
80 PythonExpression(["'ur_controllers_' + '", prefix, "' + '.yaml'"]),
81 ]
82 )
83
84 robot_description_content = Command(
85 [
86 PathJoinSubstitution([FindExecutable(name="xacro")]),
87 " ",
88 PathJoinSubstitution(
89 [FindPackageShare(description_package), "urdf", description_file]
90 ),
91 " ",
92 "safety_limits:=",
93 safety_limits,
94 " ",
95 "safety_pos_margin:=",
96 safety_pos_margin,
97 " ",
98 "safety_k_position:=",
99 safety_k_position,
100 " ",
101 "name:=",
102 prefixvalue,
103 " ",
104 "ur_type:=",
105 ur_type,
106 " ",
107 "prefix:=",
108 prefix,
109 " ",
110 "sim_gazebo:=true",
111 " ",
112 "simulation_controllers:=",
113 initial_joint_controllers,
114 " x:=",
115 x,
116 " y:=",
117 y,
118 " z:=",
119 z,
120 ]
121 )
122 robot_description = {"robot_description": robot_description_content}
123
124 xterm_prefix = "xterm -xrm 'XTerm*scrollBar: true' -xrm 'xterm*rightScrollBar: true' -hold -sl 10000 -geometry 1000x600 -e"
125
126 robot_state_publisher_node = Node(
127 package="robot_state_publisher",
128 executable="robot_state_publisher",
129 # name="robot_state_publisher", # + "_" + prefix.perform(context),
130 name="robot_state_publisher" + "_" + prefixvalue,
131 output="both",
132 parameters=[{"use_sim_time": True}, robot_description],
133 prefix=xterm_prefix,
134 remappings=[
135 ("joint_states", "/joint_state_broadcaster_" + prefixvalue + "/joint_states"),
136 ("robot_description", "robot_description" + "_" + prefix.perform(context)),
137 ],
138 )
139
140 controller_manager_name = "/" + prefix.perform(context) + "_controller_manager"
141
142 joint_state_broadcaster_spawner = Node(
143 package="controller_manager",
144 executable="spawner",
145 # name="spawn_joint_state_broadcaster",
146 prefix=xterm_prefix,
147 arguments=[
148 "joint_state_broadcaster" + "_" + prefix.perform(context),
149 "--controller-manager",
150 controller_manager_name,
151 ],
152 # arguments=["joint_state_broadcaster", "--controller-manager", "/controller_manager"],
153 )
154
155 # There may be other controllers of the joints, but this is the initially-started one
156 initial_joint_controller_spawner_started = Node(
157 package="controller_manager",
158 executable="spawner",
159 # name="spawn_trajectory_controller",
160 prefix=xterm_prefix,
161 arguments=[initial_joint_controller, "-c", controller_manager_name],
162 condition=IfCondition(start_joint_controller),
163 )
164 initial_joint_controller_spawner_stopped = Node(
165 package="controller_manager",
166 executable="spawner",
167 # name="spawn_trajectory_controller",
168 prefix=xterm_prefix,
169 arguments=[initial_joint_controller, "-c", controller_manager_name, "--stopped"],
170 condition=UnlessCondition(start_joint_controller),
171 )
172
173 # Spawn robot
174 gazebo_spawn_robot = Node(
175 package="gazebo_ros",
176 executable="spawn_entity.py",
177 name="spawn_ur",
178 arguments=[
179 "-entity",
180 prefixvalue,
181 "-topic",
182 "robot_description" + "_" + prefix.perform(context),
183 "-x",
184 "0.0",
185 "-y",
186 "0",
187 "-z",
188 "0",
189 ],
190 output="screen",
191 )
192
193 nodes_to_start = [
194 robot_state_publisher_node,
195 joint_state_broadcaster_spawner,
196 initial_joint_controller_spawner_stopped,
197 initial_joint_controller_spawner_started,
198 gazebo_spawn_robot,
199 ]
200
201 return nodes_to_start
