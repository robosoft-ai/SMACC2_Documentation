58 void loadWaypointsFromYaml(WaypointNavigator * waypointsNavigator)
59 {
60 // if it is the first time and the waypoints navigator is not configured
61 std::string planfilepath;
62
63 RCLCPP_INFO_STREAM(getLogger(), "Reasing parameter file from node: " << getNode()->get_name());
64 getNode()->declare_parameter("waypoints_plan");
65 if (getNode()->get_parameter("waypoints_plan", planfilepath))
66 {
67 std::string package_share_directory =
68 ament_index_cpp::get_package_share_directory("sm_dance_bot");
69 boost::replace_all(planfilepath, "$(pkg_share)", package_share_directory);
70 waypointsNavigator->loadWayPointsFromFile(planfilepath);
71 RCLCPP_INFO(getLogger(), "waypoints plan: %s", planfilepath.c_str());
72 }
73 else
74 {
75 RCLCPP_ERROR(getLogger(), "waypoints plan file not found: NONE");
