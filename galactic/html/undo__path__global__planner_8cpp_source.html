140void UndoPathGlobalPlanner::publishGoalMarker(
141 const geometry_msgs::msg::Pose & pose, double r, double g, double b)
142{
143 double phi = tf2::getYaw(pose.orientation);
144
145 visualization_msgs::msg::Marker marker;
146 marker.header.frame_id = this->costmap_ros_->getGlobalFrameID();
147 marker.header.stamp = nh_->now();
148 marker.ns = "my_namespace2";
149 marker.id = 0;
150 marker.type = visualization_msgs::msg::Marker::ARROW;
151 marker.action = visualization_msgs::msg::Marker::ADD;
152 marker.scale.x = 0.1;
153 marker.scale.y = 0.3;
154 marker.scale.z = 0.1;
155 marker.color.a = 1.0;
156
157 marker.color.r = r;
158 marker.color.g = g;
159 marker.color.b = b;
160
161 geometry_msgs::msg::Point start, end;
162 start.x = pose.position.x;
163 start.y = pose.position.y;
164
165 end.x = pose.position.x + 0.5 * cos(phi);
166 end.y = pose.position.y + 0.5 * sin(phi);
167
168 marker.points.push_back(start);
169 marker.points.push_back(end);
170
171 visualization_msgs::msg::MarkerArray ma;
172 ma.markers.push_back(marker);
173
174 markersPub_->publish(ma);
175}
181void UndoPathGlobalPlanner::createDefaultUndoPathPlan(
182 const geometry_msgs::msg::PoseStamped & start, const geometry_msgs::msg::PoseStamped & /*goal*/,
183 std::vector<geometry_msgs::msg::PoseStamped> & plan)
184{
185 //------------- TRANSFORM TO GLOBAL FRAME PATH ---------------------------
186 // the forward plan might be recoreded in a different frame of the global (costmap) frame. Transform it.
187 // transform global plan to the navigation reference frame
188
189 RCLCPP_INFO_STREAM(nh_->get_logger(), "[UndoPathGlobalPlanner] Transforming forward path");
190 nav_msgs::msg::Path transformedPlan;
191 rclcpp::Duration ttol = rclcpp::Duration::from_seconds(transform_tolerance_);
192 for (auto p : lastForwardPathMsg_.poses)
193 {
194 geometry_msgs::msg::PoseStamped transformedPose;
195 p.header.stamp = nh_->now(); // otherwise we can get some time tolerance error
196 transformedPose.header.stamp = nh_->now();
197 transformedPose.header.frame_id = costmap_ros_->getGlobalFrameID();
198 nav_2d_utils::transformPose(tf_, costmap_ros_->getGlobalFrameID(), p, transformedPose, ttol);
199 transformedPlan.poses.push_back(transformedPose);
200 }
201
202 lastForwardPathMsg_ = transformedPlan;
203 //---------------------------------------------------------------------------
204
205 RCLCPP_INFO_STREAM(nh_->get_logger(), "[UndoPathGlobalPlanner] finding goal closest point");
206 int i = lastForwardPathMsg_.poses.size() - 1;
207 double linear_mindist = std::numeric_limits<double>::max();
208 int mindistindex = -1;
209 double startPoseAngle = tf2::getYaw(start.pose.orientation);
210 geometry_msgs::msg::Pose startPositionProjected;
211
212 // The goal of this code is finding the most convenient initial path pose.
213 // first, find closest linear point to the current robot position
214 // we start from the final goal, that is, the beginning of the trajectory
215 // (since this was the forward motion from the odom tracker)
216 for (auto & p : transformedPlan.poses /*| boost::adaptors::reversed*/)
217 {
218 geometry_msgs::msg::PoseStamped pose = p;
219 pose.header.frame_id = costmap_ros_->getGlobalFrameID();
220
221 double dx = pose.pose.position.x - start.pose.position.x;
222 double dy = pose.pose.position.y - start.pose.position.y;
223
224 double dist = sqrt(dx * dx + dy * dy);
225 double angleOrientation = tf2::getYaw(pose.pose.orientation);
226 double angleError = fabs(angles::shortest_angular_distance(angleOrientation, startPoseAngle));
227 if (dist <= linear_mindist)
228 {
229 mindistindex = i;
230 linear_mindist = dist;
231 startPositionProjected = pose.pose;
232
233 RCLCPP_DEBUG_STREAM(
234 nh_->get_logger(), "[UndoPathGlobalPlanner] initial start point search, NEWBEST_LINEAR= "
235 << i << ". error, linear: " << linear_mindist
236 << ", angular: " << angleError);
237 }
238 else
239 {
240 RCLCPP_DEBUG_STREAM(
241 nh_->get_logger(), "[UndoPathGlobalPlanner] initial start point search, skipped= "
242 << i << ". best linear error: " << linear_mindist
243 << ". current error, linear: " << dist << " angular: " << angleError);
244 }
245
246 i--;
247 }
248
249 double const ERROR_DISTANCE_PURE_SPINNING_FACTOR = 1.5;
250 // Concept of second pass: now we only consider a pure spinning motion in this point. We want to consume some very
251 // close angular targets, (accepting a larger linear minerror of 1.5 besterror. That is, more or less in the same
252 // point).
253
254 RCLCPP_DEBUG(nh_->get_logger(), "[UndoPathGlobalPlanner] second angular pass");
255 double angularMinDist = std::numeric_limits<double>::max();
256
257 if (mindistindex >= (int)transformedPlan.poses.size())
258 mindistindex =
259 transformedPlan.poses.size() -
260 1; // workaround, something is making a out of bound exception in poses array access
261 {
262 if (transformedPlan.poses.size() == 0)
263 {
264 RCLCPP_WARN_STREAM(nh_->get_logger(), "[UndoPathGlobalPlanner] Warning possible bug");
265 }
266
267 // ------- FULL FORWARD PASS TO FIND THE STARTING POIINT OF THE FORWARD MOTION ------
268 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "[UndoPathGlobalPlanner] second pass loop");
269 for (int i = mindistindex; i >= 0; i--)
270 {
271 // warning this index, i refers to some inverse interpretation from the previous loop,
272 // (last indexes in this path corresponds to the poses closer to our current position)
273 RCLCPP_DEBUG_STREAM(
274 nh_->get_logger(), "[UndoPathGlobalPlanner] " << i << "/" << transformedPlan.poses.size());
275 auto index = (int)transformedPlan.poses.size() - i - 1;
276 if (index < 0 || (size_t)index >= transformedPlan.poses.size())
277 {
278 RCLCPP_WARN_STREAM(
279 nh_->get_logger(),
280 "[UndoPathGlobalPlanner] this should not happen. Check implementation.");
281 break;
282 }
283 geometry_msgs::msg::PoseStamped pose =
284 transformedPlan.poses[transformedPlan.poses.size() - i - 1];
285
286 RCLCPP_DEBUG_STREAM(nh_->get_logger(), "[UndoPathGlobalPlanner] global frame");
287 pose.header.frame_id = costmap_ros_->getGlobalFrameID();
288
289 double dx = pose.pose.position.x - start.pose.position.x;
290 double dy = pose.pose.position.y - start.pose.position.y;
291
292 double dist = sqrt(dx * dx + dy * dy);
293 if (dist <= linear_mindist * ERROR_DISTANCE_PURE_SPINNING_FACTOR)
294 {
295 double angleOrientation = tf2::getYaw(pose.pose.orientation);
296 double angleError =
297 fabs(angles::shortest_angular_distance(angleOrientation, startPoseAngle));
298 if (angleError < angularMinDist)
299 {
300 angularMinDist = angleError;
301 mindistindex = i;
302 RCLCPP_DEBUG_STREAM(
303 nh_->get_logger(),
304 "[UndoPathGlobalPlanner] initial start point search (angular update), NEWBEST_ANGULAR= "
305 << i << ". error, linear: " << dist << "(" << linear_mindist << ")"
306 << ", angular: " << angleError << "(" << angularMinDist << ")");
307 }
308 else
309 {
310 RCLCPP_DEBUG_STREAM(
311 nh_->get_logger(),
312 "[UndoPathGlobalPlanner] initial start point search (angular update), skipped= "
313 << i << ". error, linear: " << dist << "(" << linear_mindist << ")"
314 << ", angular: " << angleError << "(" << angularMinDist << ")");
315 }
316 }
317 else
318 {
319 RCLCPP_DEBUG_STREAM(
320 nh_->get_logger(),
321 "[UndoPathGlobalPlanner] initial start point search (angular update) not in linear "
322 "range, skipped= "
323 << i << " linear error: " << dist << "(" << linear_mindist << ")");
324 }
325 }
326 }
327
328 // REVERSE FORWARD PASS
329 if (mindistindex != -1)
330 {
331 // plan.push_back(start);
332
333 RCLCPP_WARN_STREAM(
334 nh_->get_logger(),
335 "[UndoPathGlobalPlanner] Creating the backwards plan from odom tracker path (, "
336 << transformedPlan.poses.size() << ") poses");
337
338 RCLCPP_WARN_STREAM(
339 nh_->get_logger(), "[UndoPathGlobalPlanner] closer point to goal i="
340 << mindistindex << " (linear min dist " << linear_mindist << ")");
341
342 // copy the path at the inverse direction, but only up to the closest point to the goal in the path (for partial undoing)
343 for (int i = transformedPlan.poses.size() - 1; i >= mindistindex; i--)
344 {
345 auto & pose = transformedPlan.poses[i];
346
347 rclcpp::Time t(pose.header.stamp);
348
349 RCLCPP_INFO_STREAM(
350 nh_->get_logger(),
351 "[UndoPathGlobalPlanner] adding to plan i = " << i << " stamp:" << t.seconds());
352 plan.push_back(pose);
353 }
354 RCLCPP_WARN_STREAM(
355 nh_->get_logger(), "[UndoPathGlobalPlanner] refined plan has " << plan.size() << " points");
356 }
357 else
358 {
359 RCLCPP_ERROR_STREAM(
360 nh_->get_logger(), "[UndoPathGlobalPlanner ] undo global plan size: " << plan.size());
361 }

369nav_msgs::msg::Path UndoPathGlobalPlanner::createPlan(
370 const geometry_msgs::msg::PoseStamped & start, const geometry_msgs::msg::PoseStamped & goal)
371{
372 // -------------- BASIC CHECKS ---------------------
373
374 RCLCPP_INFO_STREAM(nh_->get_logger(), "[UndoPathGlobalPlanner] Undo global plan start ");
375 nav_msgs::msg::Path planMsg;
376 std::vector<geometry_msgs::msg::PoseStamped> & plan = planMsg.poses;
377
378 RCLCPP_INFO_STREAM(
379 nh_->get_logger(),
380 "[UndoPathGlobalPlanner] last forward path msg size: " << lastForwardPathMsg_.poses.size());
381 RCLCPP_INFO_STREAM(
382 nh_->get_logger(), "[UndoPathGlobalPlanner] last forward path frame id: "
383 << lastForwardPathMsg_.poses.front().header.frame_id);
384 RCLCPP_INFO_STREAM(
385 nh_->get_logger(), "[UndoPathGlobalPlanner] start pose frame id: " << start.header.frame_id);
386 RCLCPP_INFO_STREAM(
387 nh_->get_logger(), "[UndoPathGlobalPlanner] goal pose frame id: " << goal.header.frame_id);
388
389 if (lastForwardPathMsg_.poses.size() == 0)
390 {
391 return planMsg;
392 }
393
394 // ---------- INPUTS ACCOMMODATION -------------------
395 RCLCPP_INFO_STREAM(nh_->get_logger(), "[UndoPathGlobalPlanner] Inputs accommodation");
396 geometry_msgs::msg::PoseStamped transformedStart, transformedGoal;
397 {
398 rclcpp::Duration ttol = rclcpp::Duration::from_seconds(transform_tolerance_);
399
400 geometry_msgs::msg::PoseStamped pstart = start;
401 pstart.header.stamp = nh_->now();
402 nav_2d_utils::transformPose(
403 tf_, costmap_ros_->getGlobalFrameID(), pstart, transformedStart, ttol);
404 transformedStart.header.frame_id = costmap_ros_->getGlobalFrameID();
405
406 // geometry_msgs::msg::PoseStamped pgoal = goal;
407 // pgoal.header.stamp = nh_->now();
408 // nav_2d_utils::transformPose(tf_, costmap_ros_->getGlobalFrameID(), pgoal, transformedGoal, ttol);
409 // transformedGoal.header.frame_id = costmap_ros_->getGlobalFrameID();
410
411 //--------------- FORCE GOAL POSE----------------------------
412 RCLCPP_INFO_STREAM(nh_->get_logger(), "[UndoPathGlobalPlanner] Forced goal");
413 auto forcedGoal =
414 lastForwardPathMsg_.poses[lastForwardPathMsg_.poses.size() - 1]; // FORCE LAST POSE
415 forcedGoal.header.stamp = nh_->now();
416 nav_2d_utils::transformPose(
417 tf_, costmap_ros_->getGlobalFrameID(), forcedGoal, transformedGoal, ttol);
418 transformedGoal.header.frame_id = costmap_ros_->getGlobalFrameID();
419 }
420
421 //------------- CREATING GLOBAL PLAN -----------------------------------------------
422 RCLCPP_INFO_STREAM(nh_->get_logger(), "[UndoPathGlobalPlanner] Creating undo plan");
423 this->createDefaultUndoPathPlan(transformedStart, transformedGoal, plan);
424 planMsg.header.frame_id = this->costmap_ros_->getGlobalFrameID();
425
426 RCLCPP_INFO_STREAM(nh_->get_logger(), "[UndoPathGlobalPlanner] publishing goal markers");
427 publishGoalMarker(plan.back().pose, 1.0, 0, 1.0 /*purple color*/);
428
429 //-------- CHECKING VALID PLAN ------------------------------------
430 bool acceptedGlobalPlan = true;
431 RCLCPP_INFO_STREAM(nh_->get_logger(), "[UndoPathGlobalPlanner] valid plan checking");
432
433 auto costmap2d = this->costmap_ros_->getCostmap();
434 for (auto & p : plan)
435 {
436 unsigned int mx, my;
437 costmap2d->worldToMap(p.pose.position.x, p.pose.position.y, mx, my);
438 auto cost = costmap2d->getCost(mx, my);
439
440 if (cost >= nav2_costmap_2d::INSCRIBED_INFLATED_OBSTACLE)
441 {
442 acceptedGlobalPlan = false;
443 break;
444 }
445 }
446
447 //-------- PUBLISHING RESULTS ---------------------------------------
448 RCLCPP_INFO_STREAM(
449 nh_->get_logger(), "[UndoPathGlobalPlanner] plan publishing. size: " << plan.size());
450 planPub_->publish(planMsg);
451 if (!acceptedGlobalPlan)
452 {
453 RCLCPP_INFO(
454 nh_->get_logger(),
455 "[UndoPathGlobalPlanner] not accepted global plan because of possible collision");
456 }
457
458 RCLCPP_INFO_STREAM(
459 nh_->get_logger(), "[UndoPathGlobalPlanner] plan publishing. size: " << planMsg.poses.size());
460
461 return planMsg;
462}
