33void CbRotate::onEntry()
34{
35 double angle_increment_degree = 45.0;
36
37 if (!rotateDegree)
38 {
39 this->getCurrentState()->getParam("angle_increment_degree", angle_increment_degree);
40 }
41 else
42 {
43 angle_increment_degree = *rotateDegree;
44 }
45
46 auto plannerSwitcher = moveBaseClient_->getComponent<PlannerSwitcher>();
47 // this should work better with a coroutine and await
48 // this->plannerSwitcher_->setForwardPlanner();
49 plannerSwitcher->setDefaultPlanners();
50
51 auto p = moveBaseClient_->getComponent<cl_move_base_z::Pose>();
52 auto referenceFrame = p->getReferenceFrame();
53 auto currentPoseMsg = p->toPoseMsg();
54
55 tf2::Transform currentPose;
56 tf2::fromMsg(currentPoseMsg, currentPose);
57
58 auto odomTracker = moveBaseClient_->getComponent<odom_tracker::OdomTracker>();
59 ClMoveBaseZ::Goal goal;
60 goal.pose.header.frame_id = referenceFrame;
61 goal.pose.header.stamp = getNode()->now();
62
63 auto currentAngle = tf2::getYaw(currentPoseMsg.orientation);
64 auto targetAngle = currentAngle + angle_increment_degree * M_PI / 180.0;
65 goal.pose.pose.position = currentPoseMsg.position;
66 tf2::Quaternion q;
67 q.setRPY(0, 0, targetAngle);
68 goal.pose.pose.orientation = tf2::toMsg(q);
69
70 geometry_msgs::msg::PoseStamped stampedCurrentPoseMsg;
71 stampedCurrentPoseMsg.header.frame_id = referenceFrame;
72 stampedCurrentPoseMsg.header.stamp = getNode()->now();
73 stampedCurrentPoseMsg.pose = currentPoseMsg;
74
75 this->requiresClient(moveBaseClient_);
76 odomTracker->pushPath();
77
78 odomTracker->setStartPoint(stampedCurrentPoseMsg);
79 odomTracker->setWorkingMode(odom_tracker::WorkingMode::RECORD_PATH);
80
81 RCLCPP_INFO_STREAM(getLogger(), "current pose: " << currentPoseMsg);
82 RCLCPP_INFO_STREAM(getLogger(), "goal pose: " << goal.pose.pose);
83 moveBaseClient_->sendGoal(goal);
84}
