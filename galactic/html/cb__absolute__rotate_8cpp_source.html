44void CbAbsoluteRotate::onEntry()
45{
46 double goal_angle;
47
48 listener = std::make_shared<tf2_ros::Buffer>(getNode()->get_clock());
49 if (!this->absoluteGoalAngleDegree)
50 {
51 goal_angle = 0;
52 this->getCurrentState()->param("goal_angle", goal_angle);
53 }
54 else
55 {
56 goal_angle = *this->absoluteGoalAngleDegree;
57 }
58 RCLCPP_INFO_STREAM(getLogger(), "[CbAbsoluteRotate] Absolute yaw Angle:" << goal_angle);
59
60 auto plannerSwitcher = this->moveBaseClient_->getComponent<PlannerSwitcher>();
61 // this should work better with a coroutine and await
62 // this->plannerSwitcher_->setForwardPlanner();
63
64 if (spinningPlanner && *spinningPlanner == SpiningPlanner::PureSpinning)
65 {
66 plannerSwitcher->setPureSpinningPlanner();
67 }
68 else
69 {
70 plannerSwitcher->setDefaultPlanners();
71 }
72
73 updateTemporalBehaviorParameters(false);
74
75 auto p = moveBaseClient_->getComponent<cl_move_base_z::Pose>();
76 auto referenceFrame = p->getReferenceFrame();
77 auto currentPoseMsg = p->toPoseMsg();
78
79 ClMoveBaseZ::Goal goal;
80 goal.pose.header.frame_id = referenceFrame;
81 goal.pose.header.stamp = getNode()->now();
82
83 auto targetAngle = goal_angle * M_PI / 180.0;
84 goal.pose.pose.position = currentPoseMsg.position;
85 tf2::Quaternion q;
86 q.setRPY(0, 0, targetAngle);
87 goal.pose.pose.orientation = tf2::toMsg(q);
88
89 auto odomTracker_ = moveBaseClient_->getComponent<odom_tracker::OdomTracker>();
90 if (odomTracker_ != nullptr)
91 {
92 odomTracker_->pushPath();
93 odomTracker_->setStartPoint(p->toPoseStampedMsg());
94 odomTracker_->setWorkingMode(odom_tracker::WorkingMode::RECORD_PATH);
95 }
96
97 auto goalCheckerSwitcher = moveBaseClient_->getComponent<GoalCheckerSwitcher>();
98 goalCheckerSwitcher->setGoalCheckerId("absolute_rotate_goal_checker");
99
100 RCLCPP_INFO_STREAM(
101 getLogger(),
102 "[CbAbsoluteRotate] current pose yaw: " << tf2::getYaw(currentPoseMsg.orientation));
103 RCLCPP_INFO_STREAM(
104 getLogger(), "[CbAbsoluteRotate] goal pose yaw: " << tf2::getYaw(goal.pose.pose.orientation));
105 moveBaseClient_->sendGoal(goal);
106}

108void CbAbsoluteRotate::updateTemporalBehaviorParameters(bool undo)
109{
110 auto log = this->getLogger();
111
112 std::string nodename = "/controller_server";
113
114 auto parameters_client =
115 std::make_shared<rclcpp::AsyncParametersClient>(this->getNode(), nodename);
116
117 RCLCPP_INFO_STREAM(
118 log, "[CbAbsoluteRotate] using a parameter client to update some controller parameters: "
119 << nodename << ". Waiting service.");
120 parameters_client->wait_for_service();
121
122 RCLCPP_INFO_STREAM(log, "[CbAbsoluteRotate] Service found: " << nodename << ".");
123
124 std::string localPlannerName;
125 std::vector<rclcpp::Parameter> parameters;
126
127 // dynamic_reconfigure::DoubleParameter yaw_goal_tolerance;
128 rclcpp::Parameter yaw_goal_tolerance("goal_checker.yaw_goal_tolerance");
129
130 rclcpp::Parameter max_vel_theta, min_vel_theta;
131
132 bool isRosBasePlanner = !spinningPlanner || *spinningPlanner == SpiningPlanner::Default;
133 bool isPureSpinningPlanner = spinningPlanner && *spinningPlanner == SpiningPlanner::PureSpinning;
134
135 // SELECTING CONTROLLER AND PARAMETERS NAME
136 if (isPureSpinningPlanner)
137 {
138 localPlannerName = "PureSpinningLocalPlanner";
139 max_vel_theta = rclcpp::Parameter(localPlannerName + ".max_angular_z_speed");
140 min_vel_theta = rclcpp::Parameter(localPlannerName + ".min_vel_theta");
141 }
142 else if (isRosBasePlanner)
143 {
144 localPlannerName = "FollowPath";
145 max_vel_theta = rclcpp::Parameter(localPlannerName + ".max_vel_theta");
146 min_vel_theta = rclcpp::Parameter(localPlannerName + ".min_vel_theta");
147 }
148
149 if (!undo)
150 {
151 if (yawGoalTolerance)
152 {
153 // save old yaw tolerance
154 auto fut = parameters_client->get_parameters(
155 {localPlannerName + ".yaw_goal_tolerance"}, [&](auto futureParameters) {
156 auto params = futureParameters.get();
157 oldYawTolerance = params[0].as_double();
158 });
159
160 // make synchronous
161 fut.get();
162
163 yaw_goal_tolerance = rclcpp::Parameter("goal_checker.yaw_goal_tolerance", *yawGoalTolerance);
164 parameters.push_back(yaw_goal_tolerance);
165 RCLCPP_INFO(
166 getLogger(),
167 "[CbAbsoluteRotate] updating yaw tolerance local planner to: %lf, from previous value: "
168 "%lf ",
169 *yawGoalTolerance, this->oldYawTolerance);
170 }
171
172 if (maxVelTheta)
173 {
174 if (isRosBasePlanner)
175 {
176 // nh.getParam(nodename + "/" + localPlannerName+"/min_vel_theta", oldMinVelTheta);
177 // getCurrentState()->getParam(nodename + "/" + localPlannerName + "/max_vel_theta", oldMaxVelTheta);
178
179 // save old yaw tolerance
180 auto fut = parameters_client->get_parameters(
181 {localPlannerName + ".max_vel_theta"}, [&](auto futureParameters) {
182 auto params = futureParameters.get();
183 oldMaxVelTheta = params[0].as_double();
184 });
185
186 // make synchronous
187 fut.get();
188 }
189
190 max_vel_theta = rclcpp::Parameter(localPlannerName + ".max_vel_theta", *maxVelTheta);
191 min_vel_theta = rclcpp::Parameter(localPlannerName + ".min_vel_theta", -*maxVelTheta);
192 parameters.push_back(max_vel_theta);
193 parameters.push_back(min_vel_theta);
194
195 RCLCPP_INFO(
196 log,
197 "[CbAbsoluteRotate] updating max vel theta local planner to: %lf, from previous value: "
198 "%lf ",
199 *maxVelTheta, this->oldMaxVelTheta);
200 RCLCPP_INFO(
201 log,
202 "[CbAbsoluteRotate] updating min vel theta local planner to: %lf, from previous value: "
203 "%lf ",
204 -*maxVelTheta, this->oldMinVelTheta);
205 }
206 }
207 else
208 {
209 if (yawGoalTolerance)
210 {
211 yaw_goal_tolerance = rclcpp::Parameter("goal_checker.yaw_goal_tolerance", oldYawTolerance);
212 RCLCPP_INFO(
213 log,
214 "[CbAbsoluteRotate] restoring yaw tolerance local planner from: %lf, to previous value: "
215 "%lf ",
216 *yawGoalTolerance, this->oldYawTolerance);
217 }
218
219 if (maxVelTheta)
220 {
221 if (isRosBasePlanner)
222 {
223 max_vel_theta = rclcpp::Parameter(localPlannerName + ".max_vel_theta", oldMaxVelTheta);
224 min_vel_theta = rclcpp::Parameter(localPlannerName + ".max_vel_theta", oldMinVelTheta);
225 }
226
227 parameters.push_back(max_vel_theta);
228 parameters.push_back(min_vel_theta);
229 RCLCPP_INFO(
230 log,
231 "[CbAbsoluteRotate] restoring max vel theta local planner from: %lf, to previous value: "
232 "%lf ",
233 *maxVelTheta, this->oldMaxVelTheta);
234 RCLCPP_INFO(
235 log,
236 "[CbAbsoluteRotate] restoring min vel theta local planner from: %lf, to previous value: "
237 "%lf ",
238 -(*maxVelTheta), this->oldMinVelTheta);
239 }
240 }
241
242 if (parameters.size() > 0)
243 {
244 RCLCPP_INFO(log, "[CbAbsoluteRotate] parameters to update: ");
245 for (auto & p : parameters)
246 {
247 RCLCPP_INFO_STREAM(log, "[CbAbsoluteRotate] - " << p.get_name());
248 }
249 }
250
251 auto futureResults = parameters_client->set_parameters(parameters);
252
253 int i = 0;
254 for (auto & res : futureResults.get())
255 {
256 RCLCPP_INFO_STREAM(
257 getLogger(), "[CbAbsoluteRotate] parameter result: " << parameters[i].get_name() << "="
258 << parameters[i].as_string()
259 << ". Result: " << res.successful);
260 i++;
261 }
262
263 RCLCPP_INFO(log, "[CbAbsoluteRotate] parameters updated");
264}
