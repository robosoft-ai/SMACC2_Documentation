261 declared_arguments = []
262 # UR specific arguments
263 declared_arguments.append(
264 DeclareLaunchArgument(
265 "ur_type",
266 description="Type/series of used UR robot.",

270 declared_arguments.append(
271 DeclareLaunchArgument(

273 default_value="true",
274 description="Enables the safety limits controller if true.",

277 declared_arguments.append(
278 DeclareLaunchArgument(
279 "safety_pos_margin",
280 default_value="0.15",
281 description="The margin to lower and upper limits in the safety controller.",
282 )
283 )
284 declared_arguments.append(
285 DeclareLaunchArgument(
286 "safety_k_position",
287 default_value="20",
288 description="k-position factor in the safety controller.",
289 )
290 )
291 # General arguments
292 declared_arguments.append(
293 DeclareLaunchArgument(
294 "description_package",
295 default_value="ur_description",
296 description="Description package with robot URDF/XACRO files. Usually the argument \
297 is not set, it enables use of a custom description.",
298 )
299 )
300 declared_arguments.append(
301 DeclareLaunchArgument(
302 "description_file",
303 default_value="ur.urdf.xacro",
304 description="URDF/XACRO description file with the robot.",
305 )
306 )
307 declared_arguments.append(
308 DeclareLaunchArgument(
309 "moveit_config_package",
310 default_value="ur_moveit_config",
311 description="MoveIt config package with robot SRDF/XACRO files. Usually the argument \
312 is not set, it enables use of a custom moveit config.",
313 )
314 )
315 declared_arguments.append(
316 DeclareLaunchArgument(
317 "moveit_config_file",
318 default_value="ur.srdf.xacro",
319 description="MoveIt SRDF/XACRO description file with the robot.",
320 )
321 )
322 declared_arguments.append(
323 DeclareLaunchArgument(
324 "use_sim_time",
325 default_value="false",
326 description="Make MoveIt to use simulation time. This is needed for the trajectory planing in simulation.",
327 )
328 )
329 declared_arguments.append(
330 DeclareLaunchArgument(
331 "prefix",
332 default_value='""',
333 description="Prefix of the joint names, useful for \
334 multi-robot setup. If changed than also joint names in the controllers' configuration \
335 have to be updated.",
336 )
337 )
338 declared_arguments.append(
339 DeclareLaunchArgument("launch_rviz", default_value="true", description="Launch RViz?")
340 )
341 declared_arguments.append(
342 DeclareLaunchArgument("launch_servo", default_value="true", description="Launch Servo?")
343 )
344
