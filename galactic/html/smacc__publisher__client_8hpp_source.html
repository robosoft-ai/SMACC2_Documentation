39 template <typename MessageType>
40 void configure(std::string topicName)
41 {
42 this->topicName = topicName;
43 if (!initialized_)
44 {
45 if (!this->topicName)
46 {
47 RCLCPP_ERROR(
48 getNode()->get_logger(), "topic publisher with no topic name set. Skipping advertising.");
49 return;
50 }
51
52 if (!queueSize) queueSize = 1;
53 rclcpp::SensorDataQoS qos;
54 qos.keep_last(*queueSize);
55
56 RCLCPP_INFO_STREAM(
57 getNode()->get_logger(),
58 "[" << this->getName() << "] Client Publisher to topic: " << topicName);
59 pub_ = getNode()->create_publisher<MessageType>(*(this->topicName), qos);
60
61 this->initialized_ = true;
62 }
63 }
