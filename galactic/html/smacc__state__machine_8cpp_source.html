34ISmaccStateMachine::ISmaccStateMachine(
35 std::string stateMachineName, SignalDetector * signalDetector)
36: currentState_(nullptr), stateSeqCounter_(0)
37{
38 nh_ = rclcpp::Node::make_shared(stateMachineName); //

40 nh_->get_logger(), "Creating State Machine Base: " << nh_->get_fully_qualified_name());
41
42 signalDetector_ = signalDetector;

45 std::string runMode;
46 if (nh_->get_parameter("run_mode", runMode))

55 }
56 else

58 RCLCPP_ERROR(nh_->get_logger(), "Incorrect run_mode value: %s", runMode.c_str());
59 }
60 }
61 else
62 {
63 runMode_ = SMRunMode::DEBUG;
64 }
65}

86void ISmaccStateMachine::updateStatusMessage()
87{
88 std::lock_guard<std::recursive_mutex> lock(m_mutex_);
89
90 if (currentStateInfo_ != nullptr)
91 {
92 RCLCPP_WARN_STREAM(
93 nh_->get_logger(), "[StateMachine] setting state active "
94 << ": " << currentStateInfo_->getFullPath());
95
96 if (runMode_ == SMRunMode::DEBUG)
97 {
98 status_msg_.current_states.clear();
99 std::list<const SmaccStateInfo *> ancestorList;
100 currentStateInfo_->getAncestors(ancestorList);
101
102 for (auto & ancestor : ancestorList)
103 {
104 status_msg_.current_states.push_back(ancestor->toShortName());
105 }
106
107 status_msg_.global_variable_names.clear();
108 status_msg_.global_variable_values.clear();
109
110 for (auto entry : this->globalData_)
111 {
112 status_msg_.global_variable_names.push_back(entry.first);
113 status_msg_.global_variable_values.push_back(
114 entry.second.first()); // <- invoke to_string()
115 }
116
117 status_msg_.header.stamp = this->nh_->now();
118 stateMachineStatusPub_->publish(status_msg_);
119 }
120 }
121}

142void ISmaccStateMachine::initializeROS(std::string shortname)
143{
144 RCLCPP_WARN_STREAM(nh_->get_logger(), "State machine base creation:" << shortname);
145 // STATE MACHINE TOPICS
146 stateMachinePub_ = nh_->create_publisher<smacc2_msgs::msg::SmaccStateMachine>(
147 shortname + "/smacc/state_machine_description", 1);
148 stateMachineStatusPub_ =
149 nh_->create_publisher<smacc2_msgs::msg::SmaccStatus>(shortname + "/smacc/status", 1);
150 transitionLogPub_ = nh_->create_publisher<smacc2_msgs::msg::SmaccTransitionLogEntry>(
151 shortname + "/smacc/transition_log", 1);
152
153 // STATE MACHINE SERVICES
154 transitionHistoryService_ = nh_->create_service<smacc2_msgs::srv::SmaccGetTransitionHistory>(
155 shortname + "/smacc/transition_log_history",
156 std::bind(
157 &ISmaccStateMachine::getTransitionLogHistory, this, std::placeholders::_1,
158 std::placeholders::_2, std::placeholders::_3));
159}

172void ISmaccStateMachine::state_machine_visualization()
173{
174 std::lock_guard<std::recursive_mutex> lock(m_mutex_);
175
176 smacc2_msgs::msg::SmaccStateMachine state_machine_msg;
177 state_machine_msg.states = stateMachineInfo_->stateMsgs;
178
179 std::sort(
180 state_machine_msg.states.begin(), state_machine_msg.states.end(),
181 [](auto & a, auto & b) { return a.index < b.index; });
182 stateMachinePub_->publish(state_machine_msg);
183
184 status_msg_.header.stamp = this->nh_->now();
185 stateMachineStatusPub_->publish(status_msg_);
186}
