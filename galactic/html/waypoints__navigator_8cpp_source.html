80void WaypointNavigator::seekName(std::string name)
81{
82 bool found = false;
83
84 auto previousWaypoint = currentWaypoint_;
85
86 while (!found && currentWaypoint_ < (long)waypoints_.size())
87 {
88 auto & nextName = waypointsNames_[currentWaypoint_];
89 RCLCPP_INFO(
90 getLogger(), "[WaypointNavigator] seeking ,%ld/%ld candidate waypoint: %s", currentWaypoint_,
91 waypoints_.size(), nextName.c_str());
92 if (name == nextName)
93 {
94 found = true;
95 RCLCPP_INFO(
96 getLogger(), "[WaypointNavigator] found target waypoint: %s == %s-> found",
97 nextName.c_str(), name.c_str());
98 }
99 else
100 {
101 RCLCPP_INFO(
102 getLogger(), "[WaypointNavigator] current waypoint: %s != %s -> forward", nextName.c_str(),
103 name.c_str());
104 currentWaypoint_++;
105 }
106 }
107
108 if (found)
109 {
110 if (currentWaypoint_ >= (long)waypoints_.size() - 1)
111 currentWaypoint_ = (long)waypoints_.size() - 1;
112 }
113 else // search backwards
114 {
115 currentWaypoint_ = previousWaypoint;
116 while (!found && currentWaypoint_ > 0)
117 {
118 auto & nextName = waypointsNames_[currentWaypoint_];
119 RCLCPP_INFO(
120 getLogger(), "[WaypointNavigator] seeking , candidate waypoint: %s", nextName.c_str());
121 if (name == nextName)
122 {
123 found = true;
124 RCLCPP_INFO(
125 getLogger(), "[WaypointNavigator] found target waypoint: %s == %s-> found",
126 nextName.c_str(), name.c_str());
127 }
128 else
129 {
130 RCLCPP_INFO(
131 getLogger(), "[WaypointNavigator] current waypoint: %s != %s -> rewind", nextName.c_str(),
132 name.c_str());
133 currentWaypoint_--;
134 }
135 }
136 }
137
138 RCLCPP_INFO(
139 getLogger(), "[WaypointNavigator] seekName( %s), previous index: %ld, after index: %ld",
140 name.c_str(), previousWaypoint, currentWaypoint_);
141}

154 std::shared_ptr<rclcpp_action::ClientGoalHandle<nav2_msgs::action::NavigateToPose>>>>
155WaypointNavigator::sendNextGoal(std::optional<NavigateNextWaypointOptions> options)
156{
157 if (currentWaypoint_ >= 0 && currentWaypoint_ < (int)waypoints_.size())
158 {
159 auto & next = waypoints_[currentWaypoint_];
160
161 std::string nextName;
162 if ((long)waypointsNames_.size() > currentWaypoint_)
163 {
164 nextName = waypointsNames_[currentWaypoint_];
165 RCLCPP_INFO(getLogger(), "[WaypointNavigator] sending goal, waypoint: %s", nextName.c_str());
166 }
167 else
168 {
169 RCLCPP_INFO(getLogger(), "[WaypointNavigator] sending goal, waypoint: %ld", currentWaypoint_);
170 }
171
172 ClNav2Z::Goal goal;
173 auto p = client_->getComponent<cl_nav2z::Pose>();
174 auto pose = p->toPoseMsg();
175
176 // configuring goal
177 goal.pose.header.frame_id = p->getReferenceFrame();
178 goal.pose.header.stamp = getNode()->now();
179 goal.pose.pose = next;
180
181 auto plannerSwitcher = client_->getComponent<PlannerSwitcher>();
182 plannerSwitcher->setDefaultPlanners(false);
183 if (options && options->controllerName_)
184 {
185 RCLCPP_WARN(
186 getLogger(), "[WaypointsNavigator] override controller: %s",
187 options->controllerName_->c_str());
188
189 plannerSwitcher->setDesiredController(*options->controllerName_);
190 }
191 else
192 {
193 RCLCPP_WARN(getLogger(), "[WaypointsNavigator] Configuring default planners");
194 }
195
196 auto goalCheckerSwitcher = client_->getComponent<GoalCheckerSwitcher>();
197
198 if (options && options->goalCheckerName_)
199 {
200 RCLCPP_WARN(
201 getLogger(), "[WaypointsNavigator] override goal checker: %s",
202 options->goalCheckerName_->c_str());
203
204 goalCheckerSwitcher->setGoalCheckerId(*options->goalCheckerName_);
205 }
206 else
207 {
208 RCLCPP_WARN(getLogger(), "[WaypointsNavigator] Configuring default goal checker");
209 goalCheckerSwitcher->setGoalCheckerId("goal_checker");
210 }
211
212 plannerSwitcher->commitPublish();
213
214 // publish stuff
215 // rclcpp::sleep_for(5s);
216
217 RCLCPP_INFO(getLogger(), "[WaypointsNavigator] Getting odom tracker");
218 auto odomTracker = client_->getComponent<cl_nav2z::odom_tracker::OdomTracker>();
219 if (odomTracker != nullptr)
220 {
221 RCLCPP_INFO(getLogger(), "[WaypointsNavigator] Storing path in odom tracker");
222
223 auto pathname = this->owner_->getStateMachine()->getCurrentState()->getName() + " - " +
224 getName() + " - " + nextName;
225 odomTracker->pushPath(pathname);
226 odomTracker->setStartPoint(pose);
227 odomTracker->setWorkingMode(cl_nav2z::odom_tracker::WorkingMode::RECORD_PATH);
228 }
229
230 // SEND GOAL
231 if (!succeddedNav2ZClientConnection_.connected())
232 {
233 this->succeddedNav2ZClientConnection_ =
234 client_->onSucceeded(&WaypointNavigator::onGoalReached, this);
235 this->cancelledNav2ZClientConnection_ =
236 client_->onAborted(&WaypointNavigator::onGoalCancelled, this);
237 this->abortedNav2ZClientConnection_ =
238 client_->onCancelled(&WaypointNavigator::onGoalAborted, this);
239 }
240
241 return client_->sendGoal(goal);
242 }
243 else
244 {
245 RCLCPP_WARN(
246 getLogger(),
247 "[WaypointsNavigator] All waypoints were consumed. There is no more waypoints available.");
248 }
249
250 return std::nullopt;
251}

266void WaypointNavigator::setWaypoints(const std::vector<Pose2D> & waypoints)
267{
268 waypoints_.clear();
269 waypointsNames_.clear();
270 int i = 0;
271 for (auto & p : waypoints)
272 {
273 geometry_msgs::msg::Pose pose;
274 pose.position.x = p.x_;
275 pose.position.y = p.y_;
276 pose.position.z = 0.0;
277 tf2::Quaternion q;
278 q.setRPY(0, 0, p.yaw_);
279 pose.orientation = tf2::toMsg(q);
280
281 waypoints_.push_back(pose);
282 waypointsNames_.push_back(std::to_string(i++));
283 }
284}

332void WaypointNavigator::loadWayPointsFromFile(std::string filepath)
333{
334 RCLCPP_INFO_STREAM(getLogger(), "[WaypointNavigator] Loading file:" << filepath);
335 this->waypoints_.clear();
336 std::ifstream ifs(filepath.c_str(), std::ifstream::in);
337 if (ifs.good() == false)
338 {
339 throw std::string("Waypoints file not found");
340 }
341
342 try
343 {
344#ifdef HAVE_NEW_YAMLCPP
345 YAML::Node node = YAML::Load(ifs);
346#else
347 YAML::Parser parser(ifs);
348 parser.GetNextDocument(node);
349#endif
350
351#ifdef HAVE_NEW_YAMLCPP
352 const YAML::Node & wp_node_tmp = node["waypoints"];
353 const YAML::Node * wp_node = wp_node_tmp ? &wp_node_tmp : NULL;
354#else
355 const YAML::Node * wp_node = node.FindValue("waypoints");
356#endif
357
358 if (wp_node != NULL)
359 {
360 for (unsigned int i = 0; i < wp_node->size(); ++i)
361 {
362 // Parse waypoint entries on YAML
363 geometry_msgs::msg::Pose wp;
364
365 try
366 {
367 // (*wp_node)[i]["name"] >> wp.name;
368 // (*wp_node)[i]["frame_id"] >> wp.header.frame_id;
369
370 auto wpnodei = (*wp_node)[i];
371 wp.position.x = wpnodei["position"]["x"].as<double>();
372 wp.position.y = wpnodei["position"]["y"].as<double>();
373 wp.position.z = wpnodei["position"]["z"].as<double>();
374 wp.orientation.x = wpnodei["orientation"]["x"].as<double>();
375 wp.orientation.y = wpnodei["orientation"]["y"].as<double>();
376 wp.orientation.z = wpnodei["orientation"]["z"].as<double>();
377 wp.orientation.w = wpnodei["orientation"]["w"].as<double>();
378
379 if (wpnodei["name"].IsDefined())
380 {
381 this->waypointsNames_.push_back(wpnodei["name"].as<std::string>());
382 }
383
384 this->waypoints_.push_back(wp);
385 }
386 catch (...)
387 {
388 RCLCPP_ERROR(getLogger(), "parsing waypoint file, syntax error in point %d", i);
389 }
390 }
391 RCLCPP_INFO_STREAM(getLogger(), "Parsed " << this->waypoints_.size() << " waypoints.");
392 }
393 else
394 {
395 RCLCPP_WARN_STREAM(getLogger(), "Couldn't find any waypoints in the provided yaml file.");
396 }
397 }
398 catch (const YAML::ParserException & ex)
399 {
400 RCLCPP_ERROR_STREAM(
401 getLogger(), "Error loading the Waypoints YAML file. Incorrect syntax: " << ex.what());
402 }
