48void segmentColor(const cv::Mat& inputRGB, int hueMean, int hueWindow, cv::Mat& out, int minSaturation=20, int minValue=20, bool imgShow = false)
49{
50 cv::Mat hsvInput;
51 cv::cvtColor(inputRGB, hsvInput, cv::COLOR_BGR2HSV);
52 cv::inRange(hsvInput, cv::Scalar(hueMean - hueWindow, minSaturation, minValue), cv::Scalar(hueMean + hueWindow, 255, 255), out);
53 cv::threshold(out, out, 100, 255, cv::THRESH_BINARY);
54
55 if(imgShow)
56 cv::imshow("segment", out);
57}

174int
175encoding2mat_type(const std::string & encoding)
176{
177 if (encoding == "mono8") {
178 return CV_8UC1;
179 } else if (encoding == "bgr8") {
180 return CV_8UC3;
181 } else if (encoding == "mono16") {
182 return CV_16SC1;
183 } else if (encoding == "rgba8") {
184 return CV_8UC4;
185 } else if (encoding == "bgra8") {
186 return CV_8UC4;
187 } else if (encoding == "32FC1") {
188 return CV_32FC1;
189 } else if (encoding == "rgb8") {
190 return CV_8UC3;
191 } else if (encoding == "yuv422") {
192 return CV_8UC2;
193 } else {
194 throw std::runtime_error("Unsupported encoding type");
195 }
196}

249void main_ros_loop(int argc, char** argv)
250{
251 rclcpp::init(argc, argv);
252 rclcpp::Node::SharedPtr nh = rclcpp::Node::make_shared("opencv_perception_node");
253
254 RCLCPP_INFO(nh->get_logger(), "opencv perception node started");
255 detectionPub = nh->create_publisher<std_msgs::msg::Int32>("detected_color", 1);
256 imageSub = nh->create_subscription<sensor_msgs::msg::Image>("/image_raw", rclcpp::QoS(1).best_effort(), callback);
257 debugImagePub = nh->create_publisher<sensor_msgs::msg::Image>("/opencv_debug_image", rclcpp::QoS(1).best_effort());
258
259 rclcpp::Rate r(10);
260
261 while (rclcpp::ok())
262 {
263 update();
264 rclcpp::spin_some(nh);
265 r.sleep();
266 }
267
268}
