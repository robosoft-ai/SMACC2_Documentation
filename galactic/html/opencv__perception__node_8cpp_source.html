48void segmentColor(const cv::Mat& inputRGB, int hueMean, int hueWindow, cv::Mat& out)
49{
50 cv::Mat hsvInput;
51 cv::cvtColor(inputRGB, hsvInput, cv::COLOR_BGR2HSV);
52 cv::inRange(hsvInput, cv::Scalar(hueMean - hueWindow, 20, 20), cv::Scalar(hueMean + hueWindow, 255, 255), out);
53 cv::threshold(out, out, 100, 255, cv::THRESH_BINARY);
54}

140int
141encoding2mat_type(const std::string & encoding)
142{
143 if (encoding == "mono8") {
144 return CV_8UC1;
145 } else if (encoding == "bgr8") {
146 return CV_8UC3;
147 } else if (encoding == "mono16") {
148 return CV_16SC1;
149 } else if (encoding == "rgba8") {
150 return CV_8UC4;
151 } else if (encoding == "bgra8") {
152 return CV_8UC4;
153 } else if (encoding == "32FC1") {
154 return CV_32FC1;
155 } else if (encoding == "rgb8") {
156 return CV_8UC3;
157 } else if (encoding == "yuv422") {
158 return CV_8UC2;
159 } else {
160 throw std::runtime_error("Unsupported encoding type");
161 }
162}

171void callback(const sensor_msgs::msg::Image& img)
172{
173 cv::Mat image = to_cv_mat(img);
174 sensor_msgs::msg::Image outimg = img;
175 auto outmat = to_cv_mat(outimg);
176
177 int detectedColor = 0;
178 if (testRed(image, outmat) > 0)
179 {
180 detectedColor = 1;
181 }
182
183 if (testGreen(image, outmat) > 0)
184 {
185 detectedColor = 2;
186 }
187
188 if (testBlue(image, outmat) > 0)
189 {
190 detectedColor = 3;
191 }
192
193 std_msgs::msg::Int32 detectedColorMsg;
194 detectedColorMsg.data = detectedColor;
195 detectionPub->publish(detectedColorMsg);
196
197
198 outimg.header = img.header;
199 debugImagePub->publish(outimg);
200}
