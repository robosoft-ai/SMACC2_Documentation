51void segmentColor(const cv::Mat& inputRGB, int hueMean, int hueWindow, cv::Mat& out, int minSaturation = 20,
52 int minValue = 20, bool imgShow = false)
53{
54 cv::Mat hsvInput;
55 cv::cvtColor(inputRGB, hsvInput, cv::COLOR_RGB2HSV);
56 cv::inRange(hsvInput, cv::Scalar(hueMean - hueWindow, minSaturation, minValue),
57 cv::Scalar(hueMean + hueWindow, 255, 255), out);
58 cv::threshold(out, out, 100, 255, cv::THRESH_BINARY);
59
60 if (imgShow)
61 {
62 cv::imshow("segment", out);
63 cv::waitKey();
64 }
65}

97 foundCount++;

102 cv::rectangle(debugImage, r, cv::Scalar(255, 0, 0), 1);

105 sm_husky_barrel_search_1::msg::DetectedObject detectedObject;

252int encoding2mat_type(const std::string& encoding)
253{
254 if (encoding == "mono8")
255 {
256 return CV_8UC1;
257 }
258 else if (encoding == "bgr8")
259 {
260 return CV_8UC3;
261 }
262 else if (encoding == "mono16")
263 {
264 return CV_16SC1;
265 }
266 else if (encoding == "rgba8")
267 {
268 return CV_8UC4;
269 }
270 else if (encoding == "bgra8")
271 {
272 return CV_8UC4;
273 }
274 else if (encoding == "32FC1")
275 {
276 return CV_32FC1;
277 }
278 else if (encoding == "rgb8")
279 {
280 return CV_8UC3;
281 }
282 else if (encoding == "yuv422")
283 {
284 return CV_8UC2;
285 }
286 else
287 {
288 throw std::runtime_error("Unsupported encoding type");
289 }
290}

348void main_ros_loop(int argc, char** argv)
349{
350 rclcpp::init(argc, argv);
351 rclcpp::Node::SharedPtr nh = rclcpp::Node::make_shared("opencv_perception_node");
352
353 RCLCPP_INFO(nh->get_logger(), "opencv perception node started");
354 detectionPub = nh->create_publisher<sm_husky_barrel_search_1::msg::DetectedObjects>("detected_objects", 1);
355 imageSub = nh->create_subscription<sensor_msgs::msg::Image>("/image_raw", rclcpp::QoS(1).best_effort(), callback);
356 debugImagePub = nh->create_publisher<sensor_msgs::msg::Image>("/opencv_debug_image", rclcpp::QoS(1).best_effort());
357
358 rclcpp::Rate r(10);
359
360 while (rclcpp::ok())
361 {
362 rclcpp::spin_some(nh);
363 r.sleep();
364 }
365}
