41void SmaccAsyncClientBehavior::waitFutureIfNotFinished(
42 std::optional<std::future<int>> & threadfut, bool requestFinish)
43{
44 try
45 {
46 rclcpp::Rate r(100);

49 //bool valid = threadfut.valid();
50 if (threadfut && threadfut->valid())
