34def construct_angle_radians(loader, node):
35 """utility function to construct radian values from yaml."""
36 value = loader.construct_scalar(node)
37 try:
38 return float(value)
39 except SyntaxError:
40 raise Exception("invalid expression: %s" % value)
41
42
43def construct_angle_degrees(loader, node):
44 """utility function for converting degrees into radians from yaml."""
45 return math.radians(construct_angle_radians(loader, node))

48def load_yaml(package_name, file_path):
49 package_path = get_package_share_directory(package_name)
50 absolute_file_path = os.path.join(package_path, file_path)
51
52 try:
53 yaml.SafeLoader.add_constructor("!radians", construct_angle_radians)
54 yaml.SafeLoader.add_constructor("!degrees", construct_angle_degrees)
55 except Exception:
56 raise Exception("yaml support not available; install python-yaml")
57
58 try:
59 with open(absolute_file_path) as file:
60 return yaml.safe_load(file)
61 except OSError: # parent of IOError, OSError *and* WindowsError where available
62 return None
