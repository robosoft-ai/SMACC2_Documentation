81 void MessageCallback(const sm_husky_barrel_search_1::msg::DetectedObjects& detectedMsg)
82 {
83 int totalEnemies = 0;
84
85 int totalEnemiesFireDistance = 0;
86
87 for (auto detectedObject : detectedMsg.detected_objects)
88 {
89 size_t found = detectedObject.name.find("enemy");
90
91 if (found != std::string::npos)
92 {
93 this->postEvEnemyDetected();
94 totalEnemies++;
95
96 if(detectedObject.blob_area >= 50)
97 {
98 totalEnemiesFireDistance++;
99 }
100 }
101 }
102
103 if (totalEnemies >= 3)
104 {
105 this->postEvEnemyClusterDetected();
106 }
107
108 if (totalEnemiesFireDistance > 0)
109 {
110 RCLCPP_INFO_THROTTLE(this->getLogger(), *(getNode()->get_clock()) , 1000, "Enemies in fire distance: %d", totalEnemiesFireDistance);
111 }
112
113 if(totalEnemiesFireDistance >= 3)
114 {
115 this->postEvEnemyClusterFireDistance();
116 }
117
118 }

123 this->onMessageReceived(&ClOpenCVPerception::MessageCallback, this);
124 }
