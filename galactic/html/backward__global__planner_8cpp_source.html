119void BackwardGlobalPlanner::publishGoalMarker(
120 const geometry_msgs::msg::Pose & pose, double r, double g, double b)
121{
122 double phi = tf2::getYaw(pose.orientation);
123 visualization_msgs::msg::Marker marker;
124 marker.header.frame_id = this->costmap_ros_->getGlobalFrameID();
125 marker.header.stamp = nh_->now();
126 marker.ns = "my_namespace2";
127 marker.id = 0;
128 marker.type = visualization_msgs::msg::Marker::ARROW;
129 marker.action = visualization_msgs::msg::Marker::ADD;
130 marker.scale.x = 0.1;
131 marker.scale.y = 0.3;
132 marker.scale.z = 0.1;
133 marker.color.a = 1.0;
134 marker.color.r = r;
135 marker.color.g = g;
136 marker.color.b = b;
137
138 geometry_msgs::msg::Point start, end;
139 start.x = pose.position.x;
140 start.y = pose.position.y;
141
142 end.x = pose.position.x + 0.5 * cos(phi);
143 end.y = pose.position.y + 0.5 * sin(phi);
144
145 marker.points.push_back(start);
146 marker.points.push_back(end);
147
148 visualization_msgs::msg::MarkerArray ma;
149 ma.markers.push_back(marker);
150
151 markersPub_->publish(ma);
152}

159void BackwardGlobalPlanner::createDefaultBackwardPath(
160 const geometry_msgs::msg::PoseStamped & start, const geometry_msgs::msg::PoseStamped & goal,
161 std::vector<geometry_msgs::msg::PoseStamped> & plan)
162{
163 auto q = start.pose.orientation;
164
165 geometry_msgs::msg::PoseStamped pose;
166 pose = start;
167
168 double dx = start.pose.position.x - goal.pose.position.x;
169 double dy = start.pose.position.y - goal.pose.position.y;
170
171 double length = sqrt(dx * dx + dy * dy);
172
173 geometry_msgs::msg::PoseStamped prevState;
174 if (length > skip_straight_motion_distance_)
175 {
176 // skip initial pure spinning and initial straight motion
177 // RCLCPP_INFO(nh_->get_logger(), "1 - heading to goal position pure spinning");
178 double heading_direction = atan2(dy, dx);
179 double startyaw = tf2::getYaw(q);
180 double offset = angles::shortest_angular_distance(startyaw, heading_direction);
181 heading_direction = startyaw + offset;
182
183 prevState = cl_move_base_z::makePureSpinningSubPlan(
184 start, heading_direction, plan, puresSpinningRadStep_);
185 // RCLCPP_INFO(nh_->get_logger(), "2 - going forward keep orientation pure straight");
186
187 prevState =
188 cl_move_base_z::makePureStraightSubPlan(prevState, goal.pose.position, length, plan);
189 }
190 else
191 {
192 prevState = start;
193 }
194
195 RCLCPP_WARN_STREAM(
196 nh_->get_logger(), "[BackwardGlobalPlanner] backward global plan size: " << plan.size());
197}

204nav_msgs::msg::Path BackwardGlobalPlanner::createPlan(
205 const geometry_msgs::msg::PoseStamped & start, const geometry_msgs::msg::PoseStamped & goal)
206{
207 RCLCPP_INFO_STREAM(
208 nh_->get_logger(), "[BackwardGlobalPlanner] goal frame id: "
209 << goal.header.frame_id << " pose: " << goal.pose.position);
210 RCLCPP_INFO_STREAM(
211 nh_->get_logger(), "[BackwardGlobalPlanner] goal pose frame id: " << goal.header.frame_id);
212
213 rclcpp::Duration ttol = rclcpp::Duration::from_seconds(transform_tolerance_);
214 //---------------------------------------------------------------------
215 geometry_msgs::msg::PoseStamped transformedStart;
216 nav_2d_utils::transformPose(tf_, costmap_ros_->getGlobalFrameID(), start, transformedStart, ttol);
217 transformedStart.header.frame_id = costmap_ros_->getGlobalFrameID();
218 //---------------------------------------------------------------------
219 geometry_msgs::msg::PoseStamped transformedGoal;
220 nav_2d_utils::transformPose(tf_, costmap_ros_->getGlobalFrameID(), goal, transformedGoal, ttol);
221 transformedGoal.header.frame_id = costmap_ros_->getGlobalFrameID();
222
223 //---------------------------------------------------------------------
224 std::vector<geometry_msgs::msg::PoseStamped> plan;
225 this->createDefaultBackwardPath(transformedStart, transformedGoal, plan);
226
227 publishGoalMarker(transformedGoal.pose, 1.0, 0, 1.0);
228
229 nav_msgs::msg::Path planMsg;
230 planMsg.poses = plan;
231 planMsg.header.frame_id = this->costmap_ros_->getGlobalFrameID();
232
233 //---------------------------------------------------------------------
234 // check plan rejection if obstacle is found
235 bool acceptedGlobalPlan = true;
236
237 auto costmap2d = this->costmap_ros_->getCostmap();
238 for (auto & p : plan)
239 {
240 unsigned int mx, my;
241 costmap2d->worldToMap(p.pose.position.x, p.pose.position.y, mx, my);
242 auto cost = costmap2d->getCost(mx, my);
243
244 // static const unsigned char NO_INFORMATION = 255;
245 // static const unsigned char LETHAL_OBSTACLE = 254;
246 // static const unsigned char INSCRIBED_INFLATED_OBSTACLE = 253;
247 // static const unsigned char FREE_SPACE = 0;
248 if (cost >= nav2_costmap_2d::INSCRIBED_INFLATED_OBSTACLE)
249 {
250 acceptedGlobalPlan = false;
251 break;
252 }
253 }
254
255 if (acceptedGlobalPlan)
256 {
257 RCLCPP_WARN_STREAM(
258 nh_->get_logger(),
259 "[BackwardGlobalPlanner] backward plan request is not accepted, returning "
260 "empty path");
261 planMsg.poses.clear();
262 }
263
264 RCLCPP_WARN_STREAM(
265 nh_->get_logger(), "[BackwardGlobalPlanner] backward global plan publishing path. poses count: "
266 << planMsg.poses.size());
267 planPub_->publish(planMsg);
268
269 return planMsg;
270}
