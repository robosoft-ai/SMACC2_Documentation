59 void Pose::waitTransformUpdate(rclcpp::Rate r)
60 {
61  bool found = false;
62  RCLCPP_INFO(getLogger(), "[Pose Component] waitTransformUpdate");
63  while (rclcpp::ok() && !found)
64  {
65  tf2::Stamped<tf2::Transform> transform;
66  try
67  {
68  {
69  RCLCPP_INFO_THROTTLE(
70  getLogger(), *(getNode()->get_clock()), 1000,
71  "[Pose Component] waiting transform %s -> %s", referenceFrame_.c_str(),
72  poseFrameName_.c_str());
73  std::lock_guard<std::mutex> lock(listenerMutex_);
74  auto transformstamped =
75  tfBuffer_->lookupTransform(referenceFrame_, poseFrameName_, getNode()->now());
76  tf2::fromMsg(transformstamped, transform);
77  }
78 
79  {
80  std::lock_guard<std::mutex> guard(m_mutex_);
81  tf2::toMsg(transform, this->pose_.pose);
82  this->pose_.header.stamp = tf2_ros::toRclcpp(transform.stamp_);
83  found = true;
84  this->isInitialized = true;
85  }
86  }
87  catch (tf2::TransformException & ex)
88  {
89  RCLCPP_ERROR_STREAM_THROTTLE(
90  getLogger(), *(getNode()->get_clock()), 1000,
91  "[Component pose] (" << poseFrameName_ << "/[" << referenceFrame_
92  << "] ) is failing on pose update : " << ex.what());
93  }
94 
95  r.sleep();
96  }
97  RCLCPP_INFO(getLogger(), "[Pose Component] waitTransformUpdate -> pose found!");
98 }
