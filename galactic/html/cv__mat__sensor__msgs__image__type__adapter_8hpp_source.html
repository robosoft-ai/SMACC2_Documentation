234template<>
235struct rclcpp::TypeAdapter<image_tools::ROSCvMatContainer, sensor_msgs::msg::Image>
236{
237 using is_specialized = std::true_type;
238 using custom_type = image_tools::ROSCvMatContainer;
239 using ros_message_type = sensor_msgs::msg::Image;
240
241 static
242 void
243 convert_to_ros_message(
244 const custom_type & source,
245 ros_message_type & destination)
246 {
247 destination.height = source.cv_mat().rows;
248 destination.width = source.cv_mat().cols;
249 switch (source.cv_mat().type()) {
250 case CV_8UC1:
251 destination.encoding = "mono8";
252 break;
253 case CV_8UC3:
254 destination.encoding = "bgr8";
255 break;
256 case CV_16SC1:
257 destination.encoding = "mono16";
258 break;
259 case CV_8UC4:
260 destination.encoding = "rgba8";
261 break;
262 default:
263 throw std::runtime_error("unsupported encoding type");
264 }
265 destination.step = static_cast<sensor_msgs::msg::Image::_step_type>(source.cv_mat().step);
266 size_t size = source.cv_mat().step * source.cv_mat().rows;
267 destination.data.resize(size);
268 memcpy(&destination.data[0], source.cv_mat().data, size);
269 destination.header = source.header();
270 }
271
272 static
273 void
274 convert_to_custom(
275 const ros_message_type & source,
276 custom_type & destination)
277 {
278 destination = image_tools::ROSCvMatContainer(source);
279 }
280};
